// ── Carica SOLO le impostazioni (no DOM) ──
function loadPersistedSettings() {
  try {
    const s = JSON.parse(localStorage.getItem('mfb_settings') || '{}');
    if (s.apiKey)                STATE.apiKey                = s.apiKey;
    if (s.model)                 STATE.model                 = s.model;
    if (s.threshUniversal)       STATE.threshUniversal       = +s.threshUniversal;
    if (s.threshProfile)         STATE.threshProfile         = +s.threshProfile;
    if (s.smallSampleCorrection) STATE.smallSampleCorrection = s.smallSampleCorrection;
    STATE.sessions = JSON.parse(localStorage.getItem('mfb_sessions') || '[]');
  } catch(e) {
    console.warn('loadPersistedSettings error:', e);
  }
}

// ── Ripristina STATE completo DOPO che il DOM è pronto ──
function restoreStateFromStorage() {
  try {
    const raw = localStorage.getItem('mfb_state');
    if (!raw) return;
    const parsed = JSON.parse(raw);

    if (parsed.allArgs      && Object.keys(parsed.allArgs).length)
      STATE.allArgs      = parsed.allArgs;
    if (parsed.sampleCounts && Object.keys(parsed.sampleCounts).length)
      STATE.sampleCounts = parsed.sampleCounts;
    if (parsed.selectedMateria) STATE.selectedMateria = parsed.selectedMateria;
    if (parsed.selectedCdl)     STATE.selectedCdl     = parsed.selectedCdl;

    // Ricrea file fittizi per i filtri
    if (parsed.allArgs) {
      Object.keys(parsed.allArgs).forEach(materia => {
        Object.keys(parsed.allArgs[materia]).forEach(cdl => {
          if (!STATE.files.find(f => f.materia===materia && f.cdl===cdl)) {
            STATE.files.push({
              name:'_restored_'+materia+'_'+cdl,
              size:0, status:'done',
              cdl, materia, file:null, text:'', args:[]
            });
          }
        });
      });
      updateFreqFilters();
      updateUploadStats();
      document.getElementById('uploadStats').style.display = 'grid';
      document.getElementById('countUpload').textContent   =
        STATE.files.filter(f=>f.status==='done').length;
      document.getElementById('countReview').textContent   =
        STATE.files.filter(f=>f.status==='done').length;
    }

    // Ripristina framework se presente
    if (parsed.framework) {
      STATE.framework = parsed.framework;
      document.getElementById('exportContent').style.display = 'block';
      document.getElementById('exportEmpty').style.display   = 'none';
      // renderBuilder() usa i campi HTML → richiamare solo quando
      // l'utente naviga al Builder, non subito
    }

    if (Object.keys(STATE.allArgs).length) {
      showToast('✅ Sessione precedente ripristinata');
    }
  } catch(e) {
    console.warn('restoreStateFromStorage error:', e);
    localStorage.removeItem('mfb_state');
  }
}

// ════════════════════════════════════════════════════════════════
//  AUTO-SAVE  — C1 FIX: nessuna lettura di campi HTML opzionali
//  Legge i campi solo se il Builder è visibile (display!=none)
// ════════════════════════════════════════════════════════════════
function autoSaveState() {
  try {
    // Sincronizza campi Builder → STATE solo se il Builder è montato
    if (STATE.framework) {
      const builderVisible =
        document.getElementById('builderContent').style.display !== 'none';
      if (builderVisible) {
        const v = id => document.getElementById(id)?.value?.trim() || '';
        STATE.framework.id               = v('fwId')               || STATE.framework.id;
        STATE.framework.name             = v('fwName')             || STATE.framework.name;
        STATE.framework.subject          = v('fwSubject')          || STATE.framework.subject;
        STATE.framework.context_strategy = v('fwContextStrategy');
      }
    }
    const toSave = {
      allArgs:         STATE.allArgs,
      sampleCounts:    STATE.sampleCounts,
      framework:       STATE.framework,
      selectedMateria: STATE.selectedMateria,
      selectedCdl:     STATE.selectedCdl
    };
    localStorage.setItem('mfb_state', JSON.stringify(toSave));
  } catch(e) {
    console.warn('autoSaveState error:', e);
  }
}

}

    if (STATE.files.find(x => x.name===f.name && x.cdl===STATE.selectedCdl)) return;

  updateUploadStats();
  document.getElementById('uploadStats').style.display = 'grid';
}

async function startProcessing() {
  if (!STATE.apiKey) {
    showToast('⚠️ Inserisci la chiave API nelle impostazioni','warning');
    document.getElementById('btnOpenSettings').click(); return;
  }
  const pending = STATE.files.filter(f =>
  f.status==='pending' &&
  f.materia===STATE.selectedMateria
);
  if (!pending.length) return;

  document.getElementById('processingProgress').style.display = 'block';
  document.getElementById('btnProcess').disabled = true;

  for (let i=0; i<pending.length; i++) {
    const f = pending[i];
    f.status = 'processing';
    renderFileList();
    updateProgress(i, pending.length, `Elaborazione: ${f.name}`);
    try {
      f.text = await extractTextFromPdf(f.file);
      f.args = await extractArgsWithAI(f.text, f.materia, f.cdl);
      f.status = 'done';
      aggregateArgs(f);
    } catch(e) {
      f.status = 'error';
      console.error('Errore su', f.name, e);
    }
    renderFileList();
    updateUploadStats();
  }

  updateProgress(pending.length, pending.length, 'Elaborazione completata!');
  document.getElementById('countReview').textContent =
    STATE.files.filter(f=>f.status==='done').length;
  showToast(`✅ Elaborati ${pending.filter(f=>f.status==='done').length} PDF`);
  updateFreqFilters();
  autoSaveState();
  saveSession();
}

function updateProgress(done, total, label) {
  const pct = total>0 ? Math.round((done/total)*100) : 0;
  document.getElementById('progressLabel').textContent     = label;
  document.getElementById('progressPct').textContent       = pct+'%';
  document.getElementById('mainProgressFill').style.width  = pct+'%';
}

async function extractTextFromPdf(file) {
  const buf  = await file.arrayBuffer();
  const pdf  = await pdfjsLib.getDocument({data:buf}).promise;
  let text   = '';
  for (let p=1; p<=Math.min(pdf.numPages,12); p++) {
    const page    = await pdf.getPage(p);
    const content = await page.getTextContent();
    text += content.items.map(i=>i.str).join(' ') + '\n';
  }
  return text.trim().substring(0,6000);
}

async function extractArgsWithAI(text, materiaId, cdl) {
  const materia   = MATERIE.find(m=>m.id===materiaId);
  const candidati = CANDIDATI[materiaId] || [];
  const ssd       = materia?.ssd || '';

  const systemPrompt =
    `Sei un esperto di didattica universitaria italiana specializzato nell'analisi di programmi di insegnamento.
Identifica gli argomenti disciplinari trattati, riconductoli alle categorie normalizzate fornite.
Usa ESATTAMENTE il nome della categoria se l'argomento corrisponde.
Non inserire mai lo stesso argomento due volte.
Rispondi SOLO con un array JSON di stringhe, senza spiegazioni.`;

  const userPrompt =
    `Analizza questo programma universitario di ${materia?.name||materiaId} (SSD: ${ssd}, CdL: ${cdl}).

CATEGORIE NORMALIZZATE:
${candidati.map((c,i)=>`${i+1}. ${c}`).join('\n')}

TESTO DEL PROGRAMMA:
${text}

Restituisci SOLO l'array JSON degli argomenti trovati. Ogni argomento UNA SOLA VOLTA.`;

  const resp = await fetch('https://api.openai.com/v1/chat/completions',{
    method:'POST',
    headers:{'Content-Type':'application/json','Authorization':'Bearer '+STATE.apiKey},
    body: JSON.stringify({
      model: STATE.model,
      messages:[
        {role:'system',content:systemPrompt},
        {role:'user',  content:userPrompt}
      ],
      temperature:0.1, max_tokens:800
    })
  });
  if (!resp.ok) throw new Error('API error: '+resp.status);
  const data = await resp.json();
  const raw  = data.choices[0].message.content.trim();
  let parsed = [];
  try {
    const match = raw.match(/\[[\s\S]*\]/);
    if (match) parsed = JSON.parse(match[0]);
  } catch(e) {
    parsed = raw.split('\n').map(l=>l.replace(/^[-*\d.]\s*/,'').trim()).filter(Boolean);
  }
  return [...new Set(parsed.map(a=>a.trim()).filter(Boolean))];
}

function aggregateArgs(f) {
  if (!STATE.allArgs[f.materia])              STATE.allArgs[f.materia]              = {};
  if (!STATE.allArgs[f.materia][f.cdl])       STATE.allArgs[f.materia][f.cdl]       = {};
  if (!STATE.sampleCounts[f.materia])         STATE.sampleCounts[f.materia]         = {};
  if (!STATE.sampleCounts[f.materia][f.cdl])  STATE.sampleCounts[f.materia][f.cdl]  = 0;
  STATE.sampleCounts[f.materia][f.cdl]++;
  const unique = [...new Set(f.args.map(a=>a.trim()).filter(Boolean))];
  unique.forEach(arg => {
    if (!STATE.allArgs[f.materia][f.cdl][arg])
      STATE.allArgs[f.materia][f.cdl][arg] = 0;
    STATE.allArgs[f.materia][f.cdl][arg]++;
  });
}

function getThresholds(cdls, counts) {
  const minN = Math.min(...cdls.map(c=>counts[c]||0));
  const corr = (STATE.smallSampleCorrection==='conservative' && minN<30) ? 10 : 0;
  return { u: STATE.threshUniversal-corr, p: STATE.threshProfile-corr };
}

function getArgType(arg, freqsByCdl, cdls, counts) {
  const thresh = getThresholds(cdls, counts);
  const vals   = cdls.map(c=>freqsByCdl[c]||0);
  if (vals.every(v=>v>=thresh.u)) return 'universal';
  if (vals.some(v=>v>=thresh.p))  return 'profile_dependent';
  return 'discretionary';
}

function renderFrequencyTable() {
  const materia = document.getElementById('freqMateriaFilter').value;
  if (!materia) return;
  const wrapper = document.getElementById('freqTableWrapper');
  const cdlData = STATE.allArgs[materia]      || {};
  const counts  = STATE.sampleCounts[materia] || {};
  const cdls    = Object.keys(cdlData);
  if (!cdls.length) {
    wrapper.innerHTML='<div style="text-align:center;padding:30px;color:var(--gray-3)">Nessun dato per questa materia.</div>';
    return;
  }
  const allArgSet = new Set();
  cdls.forEach(cdl=>Object.keys(cdlData[cdl]).forEach(a=>allArgSet.add(a)));
  const args   = [...allArgSet].sort();
  const thresh = getThresholds(cdls, counts);

  const thead = `<tr>
    <th>Argomento</th><th>Tipo</th>
    ${cdls.map(c=>`<th>${c}<br><span style="font-weight:400;font-size:11px">n=${counts[c]||0}</span></th>`).join('')}
  </tr>`;

  const tbody = args.map(arg => {
    const freqsByCdl = Object.fromEntries(
      cdls.map(cdl => {
        const n   = counts[cdl]||1;
        const cnt = cdlData[cdl][arg]||0;
        return [cdl, Math.round((cnt/n)*100)];
      })
    );
    const type = getArgType(arg, freqsByCdl, cdls, counts);
    const typeLabel =
      type==='universal'
        ? '<span class="badge badge-green">Universal</span>'
        : type==='profile_dependent'
        ? '<span class="badge badge-orange">Profile</span>'
        : '<span class="badge badge-gray">Discrezionale</span>';
    const cells = cdls.map(cdl => {
      const n   = counts[cdl]||1;
      const cnt = cdlData[cdl][arg]||0;
      const pct = Math.round((cnt/n)*100);
      const bc  = pct>=thresh.u?'var(--green)':pct>=thresh.p?'var(--orange)':'var(--gray-3)';
      return `<td><div class="freq-cell">
        <span class="freq-pct">${pct}%</span>
        <div style="flex:1;height:8px;border-radius:4px;background:var(--gray-2)">
          <div style="width:${Math.max(pct,2)}%;height:100%;border-radius:4px;background:${bc}"></div>
        </div>
        <span style="font-size:11px;color:var(--gray-3)">${cnt}/${n}</span>
      </div></td>`;
    }).join('');
    return `<tr><td><strong>${arg}</strong></td><td>${typeLabel}</td>${cells}</tr>`;
  }).join('');

  wrapper.innerHTML =
    `<table class="freq-table"><thead>${thead}</thead><tbody>${tbody}</tbody></table>`;

  const minN     = Math.min(...cdls.map(c=>counts[c]||0));
  const confCard = document.getElementById('confidenceCard');
  const confInfo = document.getElementById('confidenceInfo');
  confCard.style.display = 'block';
  if (minN<30) {
    confInfo.innerHTML=`<div class="callout warning"><span class="ci">⚠️</span>
      <div>Campione più piccolo: <strong>${minN} programmi</strong>. Soglie abbassate del 10%. Interpretare con cautela.</div></div>`;
  } else if (minN<50) {
    confInfo.innerHTML=`<div class="callout info"><span class="ci">ℹ️</span>
      <div>Campione più piccolo: <strong>${minN} programmi</strong>. Affidabilità <strong>sufficiente</strong>.</div></div>`;
  } else {
    confInfo.innerHTML=`<div class="callout success"><span class="ci">✅</span>
      <div>Campione <strong>solido</strong> (min ${minN} programmi). Frequenze ad alta affidabilità.</div></div>`;
  }
}

function buildFramework() {
  const materia = document.getElementById('freqMateriaFilter').value;
  if (!materia) { showToast('⚠️ Seleziona prima una materia','warning'); return; }
  const cdlData = STATE.allArgs[materia]      || {};
  const counts  = STATE.sampleCounts[materia] || {};
  const cdls    = Object.keys(cdlData);
  const info    = MATERIE.find(m=>m.id===materia);
  if (!cdls.length) { showToast('⚠️ Nessun dato di frequenza disponibile','warning'); return; }

  const allArgSet = new Set();
  cdls.forEach(cdl=>Object.keys(cdlData[cdl]).forEach(a=>allArgSet.add(a)));

  const modulesData = [...allArgSet].map(arg => {
    const freqsByCdl = Object.fromEntries(
      cdls.map(cdl => {
        const n=counts[cdl]||1, cnt=cdlData[cdl][arg]||0;
        return [cdl, Math.round((cnt/n)*100)];
      })
    );
    const type = getArgType(arg, freqsByCdl, cdls, counts);
    return {
      name: arg,
      coverage_type: type,
      core_contents: [arg+' — completare con argomenti specifici del modulo'],
      key_concepts:  [arg],
      _freqs: freqsByCdl
    };
  });

  const order = {universal:0, profile_dependent:1, discretionary:2};
  modulesData.sort((a,b)=>order[a.coverage_type]-order[b.coverage_type]);

  const cdlNames = {
    'L-24':          'L-24 Scienze e Tecniche Psicologiche',
    'L-19/LM-85bis': 'L-19/LM-85bis Scienze dell\'Educazione e della Formazione Primaria',
    'ALTRO':         'Altra classe di laurea'
  };

  const programProfiles = cdls.map(cdl => {
    const n          = counts[cdl]||0;
    const priModules = modulesData
      .filter(m=>m.coverage_type==='universal'||(m._freqs[cdl]||0)>=STATE.threshProfile)
      .map(m=>m.name);
    const secModules = modulesData
      .filter(m=>m.coverage_type==='profile_dependent'
        &&(m._freqs[cdl]||0)>=30&&(m._freqs[cdl]||0)<STATE.threshProfile)
      .map(m=>m.name);
    const priText = priModules.slice(0,4).join(', ')+(priModules.length>4?'...':'');
    return {
      name: cdlNames[cdl]||cdl,
      priority_modules:
        `[N=${n} programmi analizzati] Focus su: ${priText}. `+
        (secModules.length?`Secondari: ${secModules.slice(0,3).join(', ')}. `:'')+
        `— Completare con la logica didattica specifica per ${cdlNames[cdl]||cdl}.`
    };
  });

  const evaluationCriteria = {
    contenuti_core: 'Quali argomenti fondamentali sono presenti/assenti?',
    approccio_metodologico: 'Approccio teorico/descrittivo vs empirico/applicativo?',
    orientamento_disciplinare: 'Orientamento prevalente e scuola di pensiero?'
  };

  STATE.framework = {
    id:               materia+'_framework',
    name:             info?.name||materia,
    subject:          info?.name||materia,
    context_strategy: '',
    _meta: { ssd:info?.ssd||'', cdls, counts },
    content: {
      target_degrees:      cdls,
      evaluation_criteria: evaluationCriteria,
      program_profiles:    programProfiles,
      syllabus_modules:    modulesData
    }
  };

  showView('builderView');  // renderBuilder() chiamato da showView
  document.getElementById('exportContent').style.display = 'block';
  document.getElementById('exportEmpty').style.display   = 'none';
  autoSaveState();
  showToast('✅ Framework generato! Completa i dettagli nel Builder.');
}

function renderModulesList() {
  if (!STATE.framework) return;
  const modules = STATE.framework.content.syllabus_modules;
  const cdls    = STATE.framework._meta?.cdls||[];
  document.getElementById('modulesList').innerHTML = modules.map((m,i)=>{
    const bc =
      m.coverage_type==='universal'        ? 'var(--green)'
      : m.coverage_type==='profile_dependent' ? 'var(--orange)'
      : 'var(--gray-3)';
    const freqInfo = cdls.length&&m._freqs
      ? cdls.map(c=>`<span class="badge badge-gray" style="font-size:10px">${c}: ${m._freqs[c]||0}%</span>`).join(' ')
      : '';
    const typeBadge =
      m.coverage_type==='universal'
        ? '<span class="badge badge-green">Universal</span>'
        : m.coverage_type==='profile_dependent'
        ? '<span class="badge badge-orange">Profile-dependent</span>'
        : '<span class="badge badge-gray">Discretionary</span>';
    const contents = Array.isArray(m.core_contents) ? m.core_contents : [m.core_contents||''];
    return `
    <div class="module-card ${m.coverage_type==='profile_dependent'?'profile':m.coverage_type==='discretionary'?'discretionary':''}"
         style="border-left-color:${bc}">
      <div class="module-header">
        <div>
          <div class="module-title">${m.name}</div>
          <div style="display:flex;gap:6px;flex-wrap:wrap;margin-top:4px">
            ${typeBadge} ${freqInfo}
          </div>
        </div>
        <div style="display:flex;gap:6px">
          <button class="btn btn-secondary btn-sm" onclick="editModule(${i})">
            <i class="fas fa-edit"></i>
          </button>
          <button class="btn btn-danger btn-sm" onclick="deleteModule(${i})">
            <i class="fas fa-trash"></i>
          </button>
        </div>
      </div>
      <div class="module-tags">
        ${contents.map(c=>`<span class="module-tag">📌 ${c}</span>`).join('')}
      </div>
      ${m.key_concepts?.length
        ? `<div style="margin-top:8px;font-size:11px;color:var(--text-2)">🔑 ${m.key_concepts.join(' · ')}</div>`
        : ''}
    </div>`;
  }).join('');
}

function renderEvaluationCriteria() {
  if (!STATE.framework) return;
  const criteria = STATE.framework.content.evaluation_criteria||{};
  const list = document.getElementById('evaluationCriteriaList');
  if (!list) return;
  const entries = Object.entries(criteria);
  if (!entries.length) {
    list.innerHTML='<div style="color:var(--gray-3);font-size:13px;padding:8px 0">Nessun criterio. Clicca "Genera con AI" o aggiungi manualmente.</div>';
    return;
  }
  list.innerHTML = entries.map(([key,val],i)=>`
    <div style="display:flex;gap:10px;align-items:flex-start;margin-bottom:10px">
      <div style="flex:1">
        <input type="text" value="${key}"
          style="width:220px;font-size:12px;font-family:monospace;background:#f8f8f8;margin-bottom:4px"
          onchange="updateCriterionKey(${i},this.value)"
          placeholder="chiave_criterio">
        <textarea rows="2" style="font-size:12px;width:100%"
          onchange="updateCriterionValue('${key}',this.value)">${val}</textarea>
      </div>
      <button class="btn btn-danger btn-sm" style="margin-top:24px;flex-shrink:0"
        onclick="deleteCriterion('${key}')">
        <i class="fas fa-trash"></i>
      </button>
    </div>
  `).join('');
}

function updateCriterionKey(idx, newKey) {
  if (!STATE.framework) return;
  const entries = Object.entries(STATE.framework.content.evaluation_criteria);
  if (idx>=entries.length) return;
  const newC={};
  entries.forEach(([k,v],i)=>{ newC[i===idx?newKey:k]=v; });
  STATE.framework.content.evaluation_criteria=newC;
  refreshJsonPreview();
}
function updateCriterionValue(key,val) {
  if (STATE.framework) STATE.framework.content.evaluation_criteria[key]=val;
  refreshJsonPreview();
}
function deleteCriterion(key) {
  if (!STATE.framework) return;
  delete STATE.framework.content.evaluation_criteria[key];
  renderEvaluationCriteria(); refreshJsonPreview();
}
function addEvaluationCriterion() {
  if (!STATE.framework) return;
  STATE.framework.content.evaluation_criteria['nuovo_criterio_'+Date.now()]='';
  renderEvaluationCriteria();
}

    `).join('');
}
function updateProfileNarrative(idx,val) {

}

function editModule(idx) {
  const m = STATE.framework.content.syllabus_modules[idx];
  document.getElementById('editModuleIdx').value  = idx;
  document.getElementById('moduleName').value     = m.name;
  document.getElementById('moduleType').value     = m.coverage_type;
  document.getElementById('moduleContents').value =
    (Array.isArray(m.core_contents)?m.core_contents:[m.core_contents||'']).join('\n');
  document.getElementById('moduleKeywords').value = (m.key_concepts||[]).join('\n');
  openModal('addModuleModal');
}
function deleteModule(idx) {
  if (!confirm('Eliminare questo modulo?')) return;
  STATE.framework.content.syllabus_modules.splice(idx,1);
  renderModulesList(); refreshJsonPreview();
}
function addModule() {
  document.getElementById('editModuleIdx').value  = -1;
  document.getElementById('moduleName').value     = '';
  document.getElementById('moduleType').value     = 'universal';
  document.getElementById('moduleContents').value = '';
  document.getElementById('moduleKeywords').value = '';
  openModal('addModuleModal');
}
function saveModule() {
  const idx  = parseInt(document.getElementById('editModuleIdx').value);
  const name = document.getElementById('moduleName').value.trim();
  if (!name) { showToast('⚠️ Il nome del modulo è obbligatorio','warning'); return; }
  const m = {
    name,
    coverage_type: document.getElementById('moduleType').value,
    core_contents: document.getElementById('moduleContents').value
      .split('\n').map(l=>l.trim()).filter(Boolean),
    key_concepts:  document.getElementById('moduleKeywords').value
      .split('\n').map(l=>l.trim()).filter(Boolean),
    _freqs:{}
  };
  if (idx===-1) STATE.framework.content.syllabus_modules.push(m);
  else          STATE.framework.content.syllabus_modules[idx]=m;
  closeModal('addModuleModal');
  renderModulesList(); refreshJsonPreview(); autoSaveState();
}

// ════════════════════════════════════════════════════════════════

async function generateWithAI() {
  if (!STATE.apiKey) {
    showToast('Inserisci la chiave API nelle impostazioni', 'warning');
    openModal('settingsModal'); return;
  }
  if (!STATE.framework) {
    showToast('Genera prima il framework dalla tabella frequenze', 'warning'); return;
  }

  const btn   = document.getElementById('btnGenerateAI');
  const label = document.getElementById('aiGeneratingLabel');
  btn.disabled      = true;
  label.style.display = 'inline';

  try {
    const fw      = STATE.framework;
    const info    = MATERIE.find(function(m){ return m.id === fw.id.replace('_framework',''); });
    const cdls    = fw._meta ? fw._meta.cdls    : fw.content.target_degrees;
    const counts  = fw._meta ? fw._meta.counts  : {};
    const modules = fw.content.syllabus_modules;

    // ── Riepilogo frequenze ──
    var freqSummary = '';
    modules.forEach(function(m) {
      var freqLine = cdls.map(function(c) {
        return c + ': ' + (m._freqs ? (m._freqs[c] !== undefined ? m._freqs[c] : '?') : '?') + '%';
      }).join(', ');
      freqSummary += '- ' + m.name + ' [' + m.coverage_type + '] -> ' + freqLine + '\n';
    });

    // ── Argomenti discriminanti ──
    var discriminanti = [];
    modules.forEach(function(m) {
      if (m._freqs && cdls.length >= 2) {
        var vals  = cdls.map(function(c){ return m._freqs[c] || 0; });
        var delta = Math.max.apply(null, vals) - Math.min.apply(null, vals);
        discriminanti.push({ name: m.name, delta: delta, vals: vals });
      }
    });
    discriminanti.sort(function(a,b){ return b.delta - a.delta; });
    discriminanti = discriminanti.slice(0, 6);

    var discText = '';
    discriminanti.forEach(function(d) {
      var vstr = cdls.map(function(c,i){ return c + '=' + d.vals[i] + '%'; }).join(', ');
      discText += '- "' + d.name + '": delta ' + d.delta + '% (' + vstr + ')\n';
    });

    var cdlSample = '';
    cdls.forEach(function(c) {
      cdlSample += '- ' + c + ': ' + (counts[c] || '?') + ' programmi\n';
    });

    var totale = cdls.reduce(function(s,c){ return s + (counts[c]||0); }, 0);

    // ════════════════════════════════
    //  STEP 1 — Normalizzazione
    // ════════════════════════════════
    showToast('Step 1/2 - Normalizzazione argomenti...', 'warning');

    var argsText = '';
    modules.forEach(function(m, i) {
      var freqStr = cdls.map(function(c){
        return c + ':' + (m._freqs ? (m._freqs[c] || 0) : 0) + '%';
      }).join(', ');
      argsText += (i+1) + '. "' + m.name + '" [' + m.coverage_type + '] - ' + freqStr + '\n';
    });

    var step1System = 'Sei un esperto di didattica universitaria italiana e di analisi curricolare. '
      + 'Ricevi una lista di argomenti estratti automaticamente da programmi universitari italiani tramite OCR e AI. '
      + 'La lista contiene inevitabilmente duplicati semantici, varianti terminologiche, argomenti troppo generici o troppo specifici. '
      + 'Il tuo compito e produrre una lista normalizzata e pulita, pronta per costruire un framework editoriale professionale. '
      + 'Rispondi SOLO con un oggetto JSON valido, senza markdown, senza spiegazioni.';

    var step1User = 'Materia: ' + fw.name + ' (SSD: ' + (info ? info.ssd : '') + ')\n'
      + 'Classi di laurea analizzate:\n' + cdlSample + '\n'
      + 'LISTA ARGOMENTI GREZZI CON FREQUENZE:\n' + argsText + '\n'
      + 'ISTRUZIONI DI NORMALIZZAZIONE:\n'
      + '1. Fondi i duplicati semantici in un unico argomento normalizzato\n'
      + '2. Elimina argomenti con frequenza inferiore al 5% su TUTTI i CdL\n'
      + '3. Mantieni argomenti con frequenza pari o superiore al 5% su almeno un CdL\n'
      + '4. Usa nomi in italiano accademico standard, concisi e precisi\n'
      + '5. Ricalcola il tipo: "universal" se >= 65% su tutti i CdL, '
      + '"profile_dependent" se >= 50% su almeno un CdL, "discretionary" altrimenti\n'
      + '6. Per ogni argomento indica quali argomenti grezzi hai fuso nel campo "fused_from"\n\n'
      + 'Struttura JSON attesa:\n'
      + '{\n'
      + '  "normalized_modules": [\n'
      + '    {\n'
      + '      "name": "Nome normalizzato",\n'
      + '      "coverage_type": "universal|profile_dependent|discretionary",\n'
      + '      "freqs": {"' + (cdls[0]||'CdL1') + '": 75, "' + (cdls[1]||'CdL2') + '": 80},\n'
      + '      "fused_from": ["Argomento grezzo 1", "Argomento grezzo 2"],\n'
      + '      "key_concepts": ["concetto1", "concetto2", "concetto3"]\n'
      + '    }\n'
      + '  ],\n'
      + '  "removed": ["argomento eliminato 1"],\n'
      + '  "normalization_notes": "Breve nota sulle fusioni principali"\n'
      + '}';

    var resp1 = await fetch('https://api.openai.com/v1/chat/completions', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json', 'Authorization': 'Bearer ' + STATE.apiKey },
      body: JSON.stringify({
        model: STATE.model,
        messages: [
          { role: 'system', content: step1System },
          { role: 'user',   content: step1User   }
        ],
        temperature: 0.1,
        max_tokens: 2500
      })
    });
    if (!resp1.ok) throw new Error('API Step1 error: ' + resp1.status);
    var data1 = await resp1.json();
    var raw1  = data1.choices[0].message.content.trim();

    var step1Result;
    try {
      var match1 = raw1.match(/\{[\s\S]*\}/);
      step1Result = JSON.parse(match1 ? match1[0] : raw1);
    } catch(e) {
      throw new Error('Step 1 risposta non valida: ' + raw1.substring(0, 300));
    }

    var normalizedModules = step1Result.normalized_modules || [];
    if (!normalizedModules.length) throw new Error('Step 1 non ha prodotto moduli normalizzati');

    // ── Aggiorna moduli nel framework ──
    var order = { universal:0, profile_dependent:1, discretionary:2 };
    STATE.framework.content.syllabus_modules = normalizedModules.map(function(m) {
      var cc = (m.fused_from && m.fused_from.length > 1)
        ? ['Comprende: ' + m.fused_from.join(', ')]
        : [m.name + ' - completare con i contenuti specifici del modulo'];
      return {
        name:          m.name,
        coverage_type: m.coverage_type,
        core_contents: cc,
        key_concepts:  m.key_concepts || [m.name],
        _freqs:        m.freqs || {}
      };
    });
    STATE.framework.content.syllabus_modules.sort(function(a,b){
      return order[a.coverage_type] - order[b.coverage_type];
    });

    renderModulesList();
    showToast('Step 1 completato - ' + normalizedModules.length + ' moduli normalizzati');

    // ════════════════════════════════
    //  STEP 2 — Generazione testi
    // ════════════════════════════════
    showToast('Step 2/2 - Generazione testi framework...', 'warning');

    var moduliNorm   = STATE.framework.content.syllabus_modules;
    var universali   = moduliNorm.filter(function(m){ return m.coverage_type==='universal'; }).map(function(m){ return m.name; });
    var profileDep   = moduliNorm.filter(function(m){ return m.coverage_type==='profile_dependent'; }).map(function(m){ return m.name; });
    var discretion   = moduliNorm.filter(function(m){ return m.coverage_type==='discretionary'; }).map(function(m){ return m.name; });

    var moduliText = '';
    moduliNorm.forEach(function(m) {
      var freqStr = cdls.map(function(c){ return c + ':' + (m._freqs ? (m._freqs[c]||0) : 0) + '%'; }).join(', ');
      moduliText += '- "' + m.name + '" [' + m.coverage_type + '] ' + freqStr + '\n';
    });

    var normNotes = step1Result.normalization_notes || '';

    var cdlsStr    = cdls.join(' e ');
    var univStr    = universali.join(', ') || 'nessuno';
    var profStr    = profileDep.join(', ') || 'nessuno';
    var discStr    = discretion.join(', ') || 'nessuno';

    var step2System = 'Sei un consulente editoriale specializzato in testi universitari italiani per case editrici come Zanichelli, Il Mulino, Pearson Italia. '
      + 'Hai accesso ai dati reali di un censimento di programmi universitari italiani. '
      + 'Produci testi operativi per un framework di analisi curricolare usato da promotori editoriali sul campo. '
      + 'I testi devono essere immediatamente utilizzabili da un promotore commerciale, senza revisione editoriale. '
      + 'Scrivi in italiano professionale. Sii specifico, basati sempre sui dati reali. Evita genericita. '
      + 'Rispondi SOLO con un oggetto JSON valido, senza markdown, senza spiegazioni.';

    // ── Costruisci la lista moduli per modules_enriched ──
    var moduliEnrichedTemplate = '';
    moduliNorm.forEach(function(m, i) {
      var sep = (i < moduliNorm.length - 1) ? ',' : '';
      moduliEnrichedTemplate += '    {"name": "' + m.name + '", "core_contents": "...", "key_concepts": ["...", "...", "..."]}'  + sep + '\n';
    });

    // ── Costruisci i program_profiles template ──
    var profilesTemplate = '';
    cdls.forEach(function(c, i) {
      var sep = (i < cdls.length - 1) ? ',' : '';
      profilesTemplate += '    {"name": "' + c + '", "priority_modules": "..."}'  + sep + '\n';
    });

    var step2User = 'Framework: "' + fw.name + '" (SSD: ' + (info ? info.ssd : '') + ')\n'
      + 'Campione: ' + cdls.map(function(c){ return c + ' n=' + (counts[c]||'?'); }).join(', ')
      + ' - totale ' + totale + ' programmi\n\n'
      + 'MODULI DOPO NORMALIZZAZIONE:\n'
      + 'Universal: ' + univStr + '\n'
      + 'Profile-dependent: ' + profStr + '\n'
      + 'Discretionary: ' + discStr + '\n\n'
      + 'FREQUENZE DETTAGLIATE:\n' + moduliText + '\n'
      + 'ARGOMENTI DISCRIMINANTI:\n' + discText + '\n'
      + 'NOTE NORMALIZZAZIONE: ' + normNotes + '\n\n'
      + 'ESEMPI DI RIFERIMENTO (struttura attesa da MATRIX):\n'
      + 'context_strategy BUONA: "Corso avanzato per Economisti. La discriminante e il rigore matematico. Distinguere tra approccio discorsivo e modellistico."\n'
      + 'context_strategy CATTIVA: "La materia e presente come insegnamento obbligatorio in quasi tutti i CdL..."\n'
      + 'core_contents BUONO: "Attenzione selettiva e divisa: teorie del filtro (Broadbent, Treisman), modelli a risorse limitate (Kahneman). Elaborazione controllata vs automatica."\n'
      + 'core_contents CATTIVO: "Comprende: Attenzione selettiva, Attenzione divisa"\n'
      + 'evaluation_criterion BUONO: "Il programma distingue esplicitamente tra memoria di lavoro (Baddeley) e memoria a lungo termine?"\n'
      + 'evaluation_criterion CATTIVO: "Il programma affronta in modo approfondito il tema della Memoria?"\n\n'
      + 'Genera un oggetto JSON con questa struttura ESATTA:\n'
      + '{\n'
      + '  "context_strategy": "MAX 2 righe operative. Discriminante principale tra testi di ' + fw.name + '. Differenze chiave tra ' + cdlsStr + '. Stile: Materia dove [discriminante]. Distinguere tra [profilo A] e [profilo B]."\n,'
      + '  "modules_enriched": [\n'
      + moduliEnrichedTemplate
      + '  ],\n'
      + '  "evaluation_criteria": {\n'
      + '    "chiave_snake_1": "Domanda verificabile leggendo il programma. Discrimina tra testi con stesso indice ma approcci diversi. Formato: Il programma [azione specifica]..."\n,'
      + '    "chiave_snake_2": "Seconda domanda diagnostica specifica"\n,'
      + '    "chiave_snake_3": "Terza domanda diagnostica specifica"\n,'
      + '    "chiave_snake_4": "Quarta domanda diagnostica specifica"\n,'
      + '    "chiave_snake_5": "Quinta domanda diagnostica specifica"\n'
      + '  },\n'
      + '  "program_profiles": [\n'
      + profilesTemplate
      + '  ]\n'
      + '}\n\n'
      + 'REGOLE:\n'
      + '- context_strategy: MAX 2 righe, operativa, discriminante identificata chiaramente\n'
      + '- core_contents di modules_enriched: narrativo e denso, minimo 2 righe per modulo, con teorie e autori - MAI lista di nomi\n'
      + '- evaluation_criteria: 5 criteri con chiavi snake_case descrittive, domande verificabili\n'
      + '- program_profiles: 3-4 righe operative per promotore, con dati numerici reali\n'
      + '- Restituisci SOLO il JSON, nessun testo prima o dopo';

    var resp2 = await fetch('https://api.openai.com/v1/chat/completions', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json', 'Authorization': 'Bearer ' + STATE.apiKey },
      body: JSON.stringify({
        model: STATE.model,
        messages: [
          { role: 'system', content: step2System },
          { role: 'user',   content: step2User   }
        ],
        temperature: 0.3,
        max_tokens: 2500
      })
    });
    if (!resp2.ok) throw new Error('API Step2 error: ' + resp2.status);
    var data2 = await resp2.json();
    var raw2  = data2.choices[0].message.content.trim();

    var step2Result;
    try {
      var match2 = raw2.match(/\{[\s\S]*\}/);
      step2Result = JSON.parse(match2 ? match2[0] : raw2);
    } catch(e) {
      throw new Error('Step 2 risposta non valida: ' + raw2.substring(0, 300));
    }

    // ── Applica context_strategy ──
    if (step2Result.context_strategy) {
      STATE.framework.context_strategy = step2Result.context_strategy;
      var csEl = document.getElementById('fwContextStrategy');
      if (csEl) csEl.value = step2Result.context_strategy;
    }

    // ── Applica modules_enriched ──
    if (step2Result.modules_enriched && step2Result.modules_enriched.length) {
      step2Result.modules_enriched.forEach(function(enriched, i) {
        if (STATE.framework.content.syllabus_modules[i]) {
          if (enriched.core_contents)
            STATE.framework.content.syllabus_modules[i].core_contents = [enriched.core_contents];
          if (enriched.key_concepts && enriched.key_concepts.length)
            STATE.framework.content.syllabus_modules[i].key_concepts = enriched.key_concepts;
        }
      });
      renderModulesList();
    }

    // ── Applica evaluation_criteria ──
    if (step2Result.evaluation_criteria && Object.keys(step2Result.evaluation_criteria).length) {
      var cleaned = {};
      Object.entries(step2Result.evaluation_criteria).forEach(function(entry) {
        var k = entry[0]; var v = entry[1];
        if (k && !k.startsWith('chiave_snake') && v) cleaned[k] = v;
      });
      STATE.framework.content.evaluation_criteria =
        Object.keys(cleaned).length ? cleaned : step2Result.evaluation_criteria;
      renderEvaluationCriteria();
    }

    // ── Applica program_profiles ──
    if (step2Result.program_profiles && step2Result.program_profiles.length) {
      step2Result.program_profiles.forEach(function(pp, i) {
        if (STATE.framework.content.program_profiles[i]) {
          if (pp.priority_modules)
            STATE.framework.content.program_profiles[i].priority_modules = pp.priority_modules;
          if (pp.name)
            STATE.framework.content.program_profiles[i].name = pp.name;
        }
      });
      renderProfilesList();
    }

    refreshJsonPreview();
    showToast('Framework normalizzato e completato - pronto per export!');

  } catch(e) {
    console.error('generateWithAI error:', e);
    showToast('Errore: ' + e.message, 'error');
  } finally {
    btn.disabled        = false;
    label.style.display = 'none';
    autoSaveState();
  }
}

//  JSON FINAL — C4 FIX: usa _meta e STATE come fonte primaria,
//  i campi HTML come override se presenti e non vuoti
// ════════════════════════════════════════════════════════════════
function getFinalJson() {
  // Priorità: STATE.framework (sempre disponibile)
  // I campi HTML sovrascrivono solo se il Builder è montato
  if (!STATE.framework) return null;
  const fw = STATE.framework;

  // Leggi dai campi HTML solo se esistono nel DOM e non sono vuoti
  const readField = id => {
    const el = document.getElementById(id);
    return (el && el.value.trim()) ? el.value.trim() : null;
  };

  const id               = readField('fwId')               || fw.id;
  const name             = readField('fwName')             || fw.name;
  const subject          = readField('fwSubject')          || fw.subject;
  const context_strategy = readField('fwContextStrategy')  ?? fw.context_strategy ?? '';

  const syllabus_modules = fw.content.syllabus_modules.map(m=>{
    const core_str = Array.isArray(m.core_contents)
      ? m.core_contents.filter(Boolean).join('. ')
      : (m.core_contents||'');
    const kc = (m.key_concepts||[]).map(k=>
      typeof k==='object'?(k.concetto||k.name||JSON.stringify(k)):String(k)
    ).filter(Boolean);
    return { name:m.name, core_contents:core_str, key_concepts:kc };
  });

  const evaluation_criteria = {...(fw.content.evaluation_criteria||{})};
  delete evaluation_criteria._note;

  return {
    id, name, subject, context_strategy,
    content:{
      target_degrees:   fw.content.target_degrees,
      evaluation_criteria,
      program_profiles: fw.content.program_profiles.map(p=>({
        name:p.name, priority_modules:p.priority_modules
      })),
      syllabus_modules
    }
  };
}

function refreshJsonPreview() {
  const fw = getFinalJson();
  const el = document.getElementById('jsonPreview');
  if (!fw||!el) return;
  el.textContent = JSON.stringify(fw,null,2);
}

function downloadJson() {
  // Sincronizza STATE prima di esportare
  autoSaveState();
  const fw = getFinalJson();
  if (!fw) { showToast('⚠️ Nessun framework da esportare','warning'); return; }
  const blob = new Blob([JSON.stringify(fw,null,2)],{type:'application/json'});
  const a    = document.createElement('a');
  a.href     = URL.createObjectURL(blob);
  a.download = fw.id+'.json';
  document.body.appendChild(a);
  a.click();
  document.body.removeChild(a);
  URL.revokeObjectURL(a.href);
  showToast('✅ JSON scaricato: '+fw.id+'.json');
}

  autoSaveState();

    .then(()=>showToast('✅ JSON copiato negli appunti'))
    .catch(()=>showToast('❌ Errore copia','error'));
}

// ── CSV: usa STATE.allArgs direttamente — non dipende da campi HTML ──
function downloadCsv() {
  // Cerca la materia: prima dal select frequenze, poi dallo STATE
  const selEl   = document.getElementById('freqMateriaFilter');
  const materia = (selEl?.value) || STATE.selectedMateria ||
                  Object.keys(STATE.allArgs)[0] || '';
  if (!materia || !STATE.allArgs[materia]) {
    showToast('⚠️ Nessun dato frequenze da esportare','warning'); return;
  }
  const cdlData = STATE.allArgs[materia];
  const counts  = STATE.sampleCounts[materia]||{};
  const cdls    = Object.keys(cdlData);
  const allArgs = [...new Set(cdls.flatMap(c=>Object.keys(cdlData[c])))].sort();
  const header  = ['Argomento','Tipo',
    ...cdls.map(c=>`${c} (n=${counts[c]||0}) %`)].join(',');
  const rows = allArgs.map(arg=>{
    const freqsByCdl = Object.fromEntries(
      cdls.map(cdl=>{
        const n=counts[cdl]||1;
        return [cdl, Math.round(((cdlData[cdl][arg]||0)/n)*100)];
      })
    );
    const type = getArgType(arg,freqsByCdl,cdls,counts);
    return [`"${arg}"`,type,...cdls.map(c=>freqsByCdl[c])].join(',');
  });
  const csv  = [header,...rows].join('\n');
  const blob = new Blob([csv],{type:'text/csv;charset=utf-8;'});
  const a    = document.createElement('a');
  a.href     = URL.createObjectURL(blob);
  a.download = materia+'_frequenze.csv';
  document.body.appendChild(a);
  a.click();
  document.body.removeChild(a);
  URL.revokeObjectURL(a.href);
  showToast('✅ CSV scaricato');
}

// ── PDF: C4 FIX — usa STATE.framework direttamente, non getFinalJson()
//    che richiederebbe i campi HTML montati. Chiama getFinalJson() solo
//    dopo autoSaveState() per garantire la sincronizzazione. ──
function downloadReport() {
  autoSaveState();             // sincronizza STATE dai campi HTML se visibili
  const fw = getFinalJson();   // ora legge STATE aggiornato
  if (!fw) { showToast('⚠️ Genera prima un framework','warning'); return; }

  try {
    const { jsPDF } = window.jspdf;
    const doc = new jsPDF({orientation:'portrait',unit:'mm',format:'a4'});
    const W   = 170;
    let y     = 20;

    function addText(text,x,size,color,style){
      doc.setFontSize(size);
      doc.setTextColor(...(color||[0,0,0]));
      doc.setFont('helvetica', style==='bold'?'bold':'normal');
      const lines = doc.splitTextToSize(String(text||''), W-x+20);
      lines.forEach(line=>{
        if(y>275){doc.addPage();y=20;}
        doc.text(line,x,y);
        y+=size*0.45;
      });
      y+=2;
    }

    function addSeparator(color){
      if(y>275){doc.addPage();y=20;}
      doc.setDrawColor(...(color||[0,166,81]));
      doc.setLineWidth(0.4);
      doc.line(20,y,190,y);
      y+=5;
    }

    function addSection(title){
      if(y>260){doc.addPage();y=20;}
      y+=3;
      doc.setFillColor(0,166,81);
      doc.roundedRect(20,y-4,W,8,1,1,'F');
      doc.setFontSize(10);doc.setTextColor(255,255,255);
      doc.setFont('helvetica','bold');
      doc.text(title,23,y+1);
      y+=9;
      doc.setFont('helvetica','normal');
    }

    // Header
    doc.setFillColor(0,122,61);
    doc.rect(0,0,210,28,'F');
    doc.setFontSize(18);doc.setTextColor(255,255,255);
    doc.setFont('helvetica','bold');
    doc.text('MATRIX Framework Builder',20,12);
    doc.setFontSize(11);doc.setFont('helvetica','normal');
    doc.text('Report Metodologico — '+fw.name,20,20);
    y=36;

    addText('ID: '+fw.id,20,9,[100,100,100]);
    addText('Materia: '+fw.name+'  |  Subject: '+fw.subject,20,9,[100,100,100]);
    addText('Generato: '+new Date().toLocaleDateString('it-IT')+
      '  |  Target: '+fw.content.target_degrees.join(', '),20,9,[100,100,100]);
    addSeparator();

    const counts = STATE.framework?._meta?.counts||{};
    if(Object.keys(counts).length){
      addSection('📊 Campione analizzato');
      Object.entries(counts).forEach(([cdl,n])=>addText(`${cdl}: ${n} programmi`,24,10));
    }

    if(fw.context_strategy){
      addSection('🎯 Context Strategy');
      addText(fw.context_strategy,23,10);
    }

    if(fw.content.evaluation_criteria&&Object.keys(fw.content.evaluation_criteria).length){
      addSection('🔍 Criteri di valutazione diagnostica');
      Object.entries(fw.content.evaluation_criteria).forEach(([key,val])=>{
        addText('• '+key.replace(/_/g,' '),23,10,[0,100,60],'bold');
        addText(val,27,9,[60,60,60]);
      });
    }

    if(fw.content.program_profiles?.length){
      addSection('🎓 Program Profiles per CdL');
      fw.content.program_profiles.forEach(p=>{
        addText(p.name,23,11,[0,0,0],'bold');
        addText(p.priority_modules,25,9,[60,60,60]);
        y+=3;
      });
    }

    addSection('📚 Syllabus Modules ('+fw.content.syllabus_modules.length+')');
    const internal = STATE.framework?.content?.syllabus_modules||[];
    const nU=internal.filter(m=>m.coverage_type==='universal').length;
    const nP=internal.filter(m=>m.coverage_type==='profile_dependent').length;
    const nD=internal.filter(m=>m.coverage_type==='discretionary').length;
    addText(`Universal: ${nU}  |  Profile-dependent: ${nP}  |  Discretionary: ${nD}`,23,9,[100,100,100]);
    y+=2;

    fw.content.syllabus_modules.forEach((m,i)=>{
      if(y>268){doc.addPage();y=20;}
      const im=internal[i];
      const tipo =
        im?.coverage_type==='universal'         ? '[U]'
        : im?.coverage_type==='profile_dependent' ? '[P]'
        : '[D]';
      const tc =
        im?.coverage_type==='universal'         ? [0,166,81]
        : im?.coverage_type==='profile_dependent' ? [245,158,11]
        : [156,163,175];
      doc.setFontSize(10);doc.setFont('helvetica','bold');
      doc.setTextColor(...tc);doc.text(tipo,23,y);
      doc.setTextColor(0,0,0);doc.text(m.name,31,y);
      y+=5;
      if(m.core_contents){
        const lines=doc.splitTextToSize(m.core_contents,W-15);
        doc.setFontSize(8);doc.setFont('helvetica','normal');doc.setTextColor(80,80,80);
        lines.slice(0,2).forEach(line=>{
          if(y>275){doc.addPage();y=20;}
          doc.text(line,31,y);y+=4;
        });
      }
      y+=1;
    });

    const totalPages=doc.internal.getNumberOfPages();
    for(let p=1;p<=totalPages;p++){
      doc.setPage(p);
      doc.setFontSize(8);doc.setTextColor(180,180,180);doc.setFont('helvetica','normal');
      doc.text('MATRIX Framework Builder — '+fw.name+' — pag. '+p+'/'+totalPages,20,290);
    }

    doc.save(fw.id+'_report_'+new Date().toISOString().split('T')[0]+'.pdf');
    showToast('✅ Report PDF scaricato');
  } catch(e) {
    console.error('downloadReport error:',e);
    showToast('❌ Errore generazione PDF: '+e.message,'error');
  }
}

function saveSession() {
  const session = {
    id:          Date.now(),
    date:        new Date().toLocaleDateString('it-IT'),
    materie:     [...new Set(STATE.files.map(f=>f.materia))],
    totFiles:    STATE.files.filter(f=>f.status==='done').length,
    allArgs:     STATE.allArgs,
    sampleCounts:STATE.sampleCounts
  };
  STATE.sessions.unshift(session);
  if (STATE.sessions.length>10) STATE.sessions.pop();
  localStorage.setItem('mfb_sessions',JSON.stringify(STATE.sessions));
  renderSessions();
}

function renderSessions() {
  const list = document.getElementById('sessionsList');
  if (!STATE.sessions.length) {
    list.innerHTML=`<div style="text-align:center;padding:40px;color:var(--gray-3)">
      <i class="fas fa-history" style="font-size:32px;display:block;margin-bottom:8px"></i>
      <p>Nessuna sessione salvata.</p></div>`;
    return;
  }
  list.innerHTML = STATE.sessions.map(s=>`
    <div style="display:flex;align-items:center;gap:12px;padding:14px;border-bottom:1px solid var(--gray-2)">
      <div style="flex:1">
        <div style="font-weight:700;font-size:13px">${s.date}</div>
        <div style="font-size:12px;color:var(--text-2)">
          ${s.materie.join(', ')} · ${s.totFiles} PDF elaborati
        </div>
      </div>
      <button class="btn btn-secondary btn-sm" onclick="loadSession(${s.id})">
        <i class="fas fa-folder-open"></i> Riprendi
      </button>
      <button class="btn btn-danger btn-sm" onclick="deleteSession(${s.id})">
        <i class="fas fa-trash"></i>
      </button>
    </div>
  `).join('');
}

function loadSession(id) {
  const s = STATE.sessions.find(x=>x.id===id);
  if (!s) return;
  STATE.allArgs      = s.allArgs;
  STATE.sampleCounts = s.sampleCounts;
  updateFreqFilters();
  showView('frequencyView');
  showToast('✅ Sessione ripresa');
}

function deleteSession(id) {
  STATE.sessions = STATE.sessions.filter(s=>s.id!==id);
  localStorage.setItem('mfb_sessions',JSON.stringify(STATE.sessions));
  renderSessions();
}
