function loadSettings() {
  try {
    var s = JSON.parse(localStorage.getItem('mfb_settings') || '{}');
    if (s.apiKey)                SETTINGS.apiKey                = s.apiKey;
    if (s.model)                 SETTINGS.model                 = s.model;
    if (s.threshUniversal)       SETTINGS.threshUniversal       = +s.threshUniversal;
    if (s.threshProfile)         SETTINGS.threshProfile         = +s.threshProfile;
    if (s.smallSampleCorrection) SETTINGS.smallSampleCorrection = s.smallSampleCorrection;
  } catch(e) { console.warn('loadSettings error:', e); }
}

function saveSettings() {
  SETTINGS.apiKey                = document.getElementById('settingsApiKey').value.trim();
  SETTINGS.model                 = document.getElementById('settingsModel').value;
  SETTINGS.threshUniversal       = +document.getElementById('threshUniversal').value;
  SETTINGS.threshProfile         = +document.getElementById('threshProfile').value;
  SETTINGS.smallSampleCorrection = document.getElementById('smallSampleCorrection').value;
  localStorage.setItem('mfb_settings', JSON.stringify(SETTINGS));
  closeModal('settingsModal');
  showToast('✅ Impostazioni salvate');
}

function setupModalTriggers() {
  document.getElementById('btnOpenSettings').addEventListener('click', function() {
    document.getElementById('settingsApiKey').value        = SETTINGS.apiKey;

    document.getElementById('threshProfile').value         = SETTINGS.threshProfile;
    document.getElementById('smallSampleCorrection').value = SETTINGS.smallSampleCorrection;
    openModal('settingsModal');
  });
  ['settingsModal','newProjectModal','addModuleModal'].forEach(function(id) {
    document.getElementById(id).addEventListener('click', function(e) {
      if (e.target === this) closeModal(id);
    });
  });
}

function getAllProjects() {
  var projects = [];
  for (var i = 0; i < localStorage.length; i++) {
    var key = localStorage.key(i);
    if (key && key.startsWith('mfb_project_')) {
      try {
        var p = JSON.parse(localStorage.getItem(key));
        if (p && p.id) projects.push(p);
      } catch(e) {}
    }
  }
  projects.sort(function(a,b){ return (b.updatedAt||0) - (a.updatedAt||0); });
  return projects;
}

function saveProject(project) {
  project.updatedAt = Date.now();
  localStorage.setItem('mfb_project_' + project.id, JSON.stringify(project));
}

function deleteProject(projectId) {
  localStorage.removeItem('mfb_project_' + projectId);
}

function createNewProject(name, materiaId, materiaName, materiaSsd, cdls) {
  var id = 'proj_' + Date.now();
  var project = {
    id:          id,
    name:        name,
    materiaId:   materiaId,
    materiaName: materiaName,
    materiaSsd:  materiaSsd,
    cdls:        cdls,
    createdAt:   Date.now(),
    updatedAt:   Date.now(),
    status:      'new',    // new | working | framework | exported
    files:       [],
    allArgs:     {},
    sampleCounts:{},
    framework:   null
  };
  saveProject(project);
  return project;
}

function getProjectStatus(project) {
  if (project.framework) return 'exported';
  if (project.allArgs && Object.keys(project.allArgs).length) return 'framework';
  if (project.files && project.files.filter(function(f){ return f.status==='done'; }).length)
    return 'working';
  return 'new';
}

function onCdlCheckboxChange(checkbox) {
  var id = checkbox.value;
  if (checkbox.checked) {
    if (wizardSelectedCdls.indexOf(id) < 0) wizardSelectedCdls.push(id);
  } else {
    wizardSelectedCdls = wizardSelectedCdls.filter(function(c) { return c !== id; });
  }
  renderWizardCdlSelected();
}
function filterCdlCheckboxes(query) {
  var q = query.toLowerCase().trim();
  var labels = document.querySelectorAll('#cdlCheckboxList label');
  labels.forEach(function(lbl) {
    var text = lbl.textContent.toLowerCase();
    lbl.style.display = (!q || text.includes(q)) ? '' : 'none';
  });
}
function removeCdlBadge(id) {
  wizardSelectedCdls = wizardSelectedCdls.filter(function(c) { return c !== id; });
  var checkboxes = document.querySelectorAll('#cdlCheckboxList input[type="checkbox"]');
  checkboxes.forEach(function(cb) {
    if (cb.value === id) cb.checked = false;
  });
  renderWizardCdlSelected();
}
function addCdlFromInlineSelect() {
  var sel = document.getElementById('wizardCdlSelectInline');
  if (!sel || !sel.value) { showToast('⚠️ Seleziona una classe di laurea', 'warning'); return; }
  var id = sel.value;
  if (wizardSelectedCdls.indexOf(id) >= 0) { showToast('⚠️ Già presente', 'warning'); return; }
  wizardSelectedCdls.push(id);
  sel.value = '';
  renderWizardCdlSelected();
}

function toggleWizardCdl(id) {
  var idx = wizardSelectedCdls.indexOf(id);
  if (idx >= 0) {
    wizardSelectedCdls.splice(idx, 1);
  } else {
    wizardSelectedCdls.push(id);
  }
  var btn = document.getElementById('wcdl-' + id.replace(/\//g,'_'));
  if (btn) btn.classList.toggle('selected', wizardSelectedCdls.indexOf(id) >= 0);
  renderWizardCdlSelected();
}

function updateWizardUI() {
  [1,2,3].forEach(function(i) {
    var step  = document.getElementById('wstep'  + i);
    var panel = document.getElementById('wpanel' + i);
    step.classList.toggle('active', i === wizardStep);
    step.classList.toggle('done',   i <  wizardStep);
    panel.classList.toggle('active', i === wizardStep);
  });
  var btnBack = document.getElementById('btnWizardBack');
  var btnNext = document.getElementById('btnWizardNext');
  btnBack.style.display = wizardStep > 1 ? 'inline-flex' : 'none';
  btnNext.innerHTML = wizardStep < 3
    ? 'Avanti <i class="fas fa-arrow-right"></i>'
    : '<i class="fas fa-check"></i> Crea progetto';
}

function wizardNext() {
  if (wizardStep === 1) {
    var name = document.getElementById('newProjectName').value.trim();
    if (!name) { showToast('⚠️ Inserisci un nome per il progetto', 'warning'); return; }
    wizardStep = 2;
    updateWizardUI();
  } else if (wizardStep === 2) {
    if (!wizardSelectedMateria) {
      showToast('⚠️ Seleziona una materia', 'warning'); return;
    }
    if (wizardSelectedMateria === 'custom') {
      var cname = document.getElementById('customMateriaName').value.trim();
      if (!cname) { showToast('⚠️ Inserisci il nome della materia', 'warning'); return; }
    }
    wizardStep = 3;
    updateWizardUI();
  } else if (wizardStep === 3) {
    if (!wizardSelectedCdls.length) {
      showToast('⚠️ Seleziona almeno una classe di laurea', 'warning'); return;
    }
    // Crea il progetto
    var pname    = document.getElementById('newProjectName').value.trim();
    var mId      = wizardSelectedMateria;
    var mInfo    = MATERIE.find(function(m){ return m.id === mId; });
    var mName    = mId === 'custom'

function autoSaveProject() {
  if (!CURRENT_PROJECT) return;

        document.getElementById('builderContent').style.display !== 'none';
      if (builderVisible) {

          return (el && el.value.trim()) ? el.value.trim() : null;
        };
        CURRENT_PROJECT.framework.id               = v('fwId')             || CURRENT_PROJECT.framework.id;
        CURRENT_PROJECT.framework.name             = v('fwName')           || CURRENT_PROJECT.framework.name;

        CURRENT_PROJECT.framework.context_strategy = v('fwContextStrategy') || CURRENT_PROJECT.framework.context_strategy || '';
      }
    }
    // Non salviamo i File object (non serializzabili) — solo metadati
    var toSave = Object.assign({}, CURRENT_PROJECT);
    toSave.files = (CURRENT_PROJECT.files || []).map(function(f) {
      return {
        name:    f.name,
        size:    f.size,
        status:  f.status,
        cdl:     f.cdl,
        materia: f.materia,
        args:    f.args || [],

      };
    });
    saveProject(toSave);
    CURRENT_PROJECT.updatedAt = toSave.updatedAt;
  } catch(e) {
    console.warn('autoSaveProject error:', e);
  }
}

function setupDropZone() {
  var dz = document.getElementById('dropZone');
  if (!dz) return;
  dz.addEventListener('dragover', function(e){
    e.preventDefault(); dz.classList.add('drag-over');
  });
  dz.addEventListener('dragleave', function(){
    dz.classList.remove('drag-over');
  });
  dz.addEventListener('drop', function(e){
    e.preventDefault();
    dz.classList.remove('drag-over');
    handleFiles(e.dataTransfer.files);
  });
}

function handleFiles(fileList) {
  if (!CURRENT_PROJECT || !CURRENT_PROJECT.selectedCdl) {
    showToast('⚠️ Seleziona prima una classe di laurea', 'warning'); return;
  }
  var cdl     = CURRENT_PROJECT.selectedCdl;
  var materia = CURRENT_PROJECT.materiaId;
  Array.from(fileList).filter(function(f){ return f.name.endsWith('.pdf'); })
    .forEach(function(f) {
      if (CURRENT_PROJECT.files.find(function(x){
        return x.name === f.name && x.cdl === cdl;
      })) return;
      CURRENT_PROJECT.files.push({
        name: f.name, size: f.size, status: 'pending',
        cdl: cdl, materia: materia,
        file: f, text: '', args: []
      });
    });
  renderFileList();
  updateBtnProcess();
  updateUploadStats();
  document.getElementById('uploadStats').style.display = 'grid';
}

async function startProcessing() {
  if (!SETTINGS.apiKey) {
    showToast('⚠️ Inserisci la chiave API nelle impostazioni', 'warning');
    document.getElementById('btnOpenSettings').click(); return;
  }
  if (!CURRENT_PROJECT) return;
  var pending = CURRENT_PROJECT.files.filter(function(f){
    return f.status==='pending' && f.materia===CURRENT_PROJECT.materiaId;
  });
  if (!pending.length) return;

  document.getElementById('processingProgress').style.display = 'block';
  document.getElementById('btnProcess').disabled = true;

  for (var i=0; i<pending.length; i++) {
    var f = pending[i];
    f.status = 'processing';
    renderFileList();
    updateProgress(i, pending.length, 'Elaborazione: ' + f.name);
    try {
      f.text = await extractTextFromPdf(f.file);
      f.args = await extractArgsWithAI(f.text, CURRENT_PROJECT.materiaId,
                                        CURRENT_PROJECT.materiaSsd, f.cdl);
      f.status = 'done';
      aggregateArgs(f);
    } catch(e) {
      f.status = 'error';
      console.error('Errore su', f.name, e);
    }
    renderFileList();
    updateUploadStats();
  }

  updateProgress(pending.length, pending.length, 'Elaborazione completata!');
  document.getElementById('countReview').textContent =
    CURRENT_PROJECT.files.filter(function(f){ return f.status==='done'; }).length;
  showToast('✅ Elaborati ' + pending.filter(function(f){ return f.status==='done'; }).length + ' PDF');
  updateReviewCdlFilter();
  autoSaveProject();
}

function updateProgress(done, total, label) {
  var pct = total>0 ? Math.round((done/total)*100) : 0;
  document.getElementById('progressLabel').textContent    = label;
  document.getElementById('progressPct').textContent      = pct + '%';
  document.getElementById('mainProgressFill').style.width = pct + '%';
}

async function extractTextFromPdf(file) {
  var buf  = await file.arrayBuffer();
  var pdf  = await pdfjsLib.getDocument({data:buf}).promise;
  var text = '';
  for (var p=1; p<=Math.min(pdf.numPages,12); p++) {
    var page    = await pdf.getPage(p);
    var content = await page.getTextContent();
    text += content.items.map(function(i){ return i.str; }).join(' ') + '\n';
  }
  return text.trim().substring(0,6000);
}

async function extractArgsWithAI(text, materiaId, materiaSsd, cdl) {
  var mInfo     = MATERIE.find(function(m){ return m.id===materiaId; });
  var mName     = mInfo ? mInfo.name : materiaId;
  var candidati = CANDIDATI[materiaId] || [];
  var ssd       = materiaSsd || (mInfo ? mInfo.ssd : '') || '';

  var systemPrompt =
    'Sei un esperto di didattica universitaria italiana specializzato nell\'analisi di programmi di insegnamento. '
    + 'Identifica gli argomenti disciplinari trattati, riconductoli alle categorie normalizzate fornite. '
    + 'Usa ESATTAMENTE il nome della categoria se l\'argomento corrisponde. '
    + 'Non inserire mai lo stesso argomento due volte. '
    + 'Rispondi SOLO con un array JSON di stringhe, senza spiegazioni.';

  var candidatiText = candidati.length
    ? 'CATEGORIE NORMALIZZATE:\n' + candidati.map(function(c,i){ return (i+1)+'. '+c; }).join('\n') + '\n\n'
    : '';

  var userPrompt =
    'Analizza questo programma universitario di ' + mName + ' (SSD: ' + ssd + ', CdL: ' + cdl + ').\n\n'
    + candidatiText
    + 'TESTO DEL PROGRAMMA:\n' + text + '\n\n'
    + 'Restituisci SOLO l\'array JSON degli argomenti trovati. Ogni argomento UNA SOLA VOLTA.';

  var resp = await fetch('https://api.openai.com/v1/chat/completions', {
    method: 'POST',
    headers: {'Content-Type':'application/json','Authorization':'Bearer '+SETTINGS.apiKey},
    body: JSON.stringify({
      model: SETTINGS.model,
      messages: [
        {role:'system', content:systemPrompt},
        {role:'user',   content:userPrompt}
      ],
      temperature: 0.1, max_tokens: 800
    })
  });
  if (!resp.ok) throw new Error('API error: ' + resp.status);
  var data  = await resp.json();
  var raw   = data.choices[0].message.content.trim();
  var parsed = [];
  try {
    var match = raw.match(/\[[\s\S]*\]/);
    if (match) parsed = JSON.parse(match[0]);
  } catch(e) {
    parsed = raw.split('\n').map(function(l){
      return l.replace(/^[-*\d.]\s*/,'').trim();
    }).filter(Boolean);
  }
  return [...new Set(parsed.map(function(a){ return a.trim(); }).filter(Boolean))];
}

function aggregateArgs(f) {
  if (!CURRENT_PROJECT) return;
  var m = CURRENT_PROJECT.materiaId;
  var c = f.cdl;
  if (!CURRENT_PROJECT.allArgs[m])         CURRENT_PROJECT.allArgs[m]         = {};
  if (!CURRENT_PROJECT.allArgs[m][c])      CURRENT_PROJECT.allArgs[m][c]      = {};
  if (!CURRENT_PROJECT.sampleCounts[m])    CURRENT_PROJECT.sampleCounts[m]    = {};
  if (!CURRENT_PROJECT.sampleCounts[m][c]) CURRENT_PROJECT.sampleCounts[m][c] = 0;
  CURRENT_PROJECT.sampleCounts[m][c]++;
  var unique = [...new Set((f.args||[]).map(function(a){ return a.trim(); }).filter(Boolean))];
  unique.forEach(function(arg) {
    if (!CURRENT_PROJECT.allArgs[m][c][arg]) CURRENT_PROJECT.allArgs[m][c][arg] = 0;
    CURRENT_PROJECT.allArgs[m][c][arg]++;
  });
}

// ════════════════════════════════════════════════
//  FREQUENZE
// ════════════════════════════════════════════════
function getThresholds(cdls, counts) {
  var minN = Math.min.apply(null, cdls.map(function(c){ return counts[c]||0; }));
  var corr = (SETTINGS.smallSampleCorrection==='conservative' && minN<30) ? 10 : 0;
  return { u: SETTINGS.threshUniversal-corr, p: SETTINGS.threshProfile-corr };
}

function getArgType(arg, freqsByCdl, cdls, counts) {
  var thresh = getThresholds(cdls, counts);
  var vals   = cdls.map(function(c){ return freqsByCdl[c]||0; });
  if (vals.every(function(v){ return v>=thresh.u; })) return 'universal';
  if (vals.some(function(v){  return v>=thresh.p; })) return 'profile_dependent';
  return 'discretionary';
}

function renderFrequencyTable() {
  if (!CURRENT_PROJECT) return;
  var materia = CURRENT_PROJECT.materiaId;
  var wrapper = document.getElementById('freqTableWrapper');
  var cdlData = (CURRENT_PROJECT.allArgs[materia])      || {};
  var counts  = (CURRENT_PROJECT.sampleCounts[materia]) || {};
  var cdls    = Object.keys(cdlData);

  if (!cdls.length) {
    wrapper.innerHTML='<div style="text-align:center;padding:30px;color:var(--gray-3)">Nessun dato disponibile. Elabora i PDF per procedere.</div>';
    return;
  }

  var allArgSet = new Set();
  cdls.forEach(function(cdl){ Object.keys(cdlData[cdl]).forEach(function(a){ allArgSet.add(a); }); });
  var args   = [...allArgSet].sort();
  var thresh = getThresholds(cdls, counts);

  var thead = '<tr><th>Argomento</th><th>Tipo</th>'
    + cdls.map(function(c){
        return '<th>' + escHtml(c) + '<br><span style="font-weight:400;font-size:11px">n=' + (counts[c]||0) + '</span></th>';
      }).join('')
    + '</tr>';

  var tbody = args.map(function(arg) {
    var freqsByCdl = {};
    cdls.forEach(function(cdl){
      var n   = counts[cdl]||1;
      var cnt = cdlData[cdl][arg]||0;
      freqsByCdl[cdl] = Math.round((cnt/n)*100);
    });
    var type = getArgType(arg, freqsByCdl, cdls, counts);
    var typeLabel =
      type==='universal'
        ? '<span class="badge badge-green">Universal</span>'
        : type==='profile_dependent'
        ? '<span class="badge badge-orange">Profile</span>'
        : '<span class="badge badge-gray">Discrezionale</span>';
    var cells = cdls.map(function(cdl){
      var n   = counts[cdl]||1;
      var cnt = cdlData[cdl][arg]||0;
      var pct = Math.round((cnt/n)*100);
      var bc  = pct>=thresh.u ? 'var(--green)' : pct>=thresh.p ? 'var(--orange)' : 'var(--gray-3)';
      return '<td><div class="freq-cell">'
           + '  <span class="freq-pct">' + pct + '%</span>'
           + '  <div style="flex:1;height:8px;border-radius:4px;background:var(--gray-2)">'
           + '    <div style="width:' + Math.max(pct,2) + '%;height:100%;border-radius:4px;background:' + bc + '"></div>'
           + '  </div>'
           + '  <span style="font-size:11px;color:var(--gray-3)">' + cnt + '/' + n + '</span>'
           + '</div></td>';
    }).join('');
    return '<tr><td><strong>' + escHtml(arg) + '</strong></td><td>' + typeLabel + '</td>' + cells + '</tr>';
  }).join('');

  wrapper.innerHTML = '<table class="freq-table"><thead>' + thead + '</thead><tbody>' + tbody + '</tbody></table>';

  // Confidence
  var minN     = Math.min.apply(null, cdls.map(function(c){ return counts[c]||0; }));
  var confCard = document.getElementById('confidenceCard');
  var confInfo = document.getElementById('confidenceInfo');
  confCard.style.display = 'block';
  if (minN < 30) {
    confInfo.innerHTML='<div class="callout warning"><span class="ci">⚠️</span><div>Campione piccolo: <strong>' + minN + ' programmi</strong>. Soglie abbassate del 10%.</div></div>';
  } else if (minN < 50) {
    confInfo.innerHTML='<div class="callout info"><span class="ci">ℹ️</span><div>Campione: <strong>' + minN + ' programmi</strong>. Affidabilità <strong>sufficiente</strong>.</div></div>';
  } else {
    confInfo.innerHTML='<div class="callout success"><span class="ci">✅</span><div>Campione <strong>solido</strong> (min ' + minN + ' programmi).</div></div>';
  }
}

function buildFramework() {
  if (!CURRENT_PROJECT) return;
  var materia = CURRENT_PROJECT.materiaId;
  var cdlData = CURRENT_PROJECT.allArgs[materia]      || {};
  var counts  = CURRENT_PROJECT.sampleCounts[materia] || {};
  var cdls    = Object.keys(cdlData);
  if (!cdls.length) { showToast('⚠️ Nessun dato di frequenza disponibile','warning'); return; }

  var allArgSet = new Set();
  cdls.forEach(function(cdl){ Object.keys(cdlData[cdl]).forEach(function(a){ allArgSet.add(a); }); });

  var modulesData = [...allArgSet].map(function(arg){
    var freqsByCdl = {};
    cdls.forEach(function(cdl){
      var n=counts[cdl]||1, cnt=cdlData[cdl][arg]||0;
      freqsByCdl[cdl] = Math.round((cnt/n)*100);
    });
    var type = getArgType(arg, freqsByCdl, cdls, counts);
    return {
      name:          arg,
      coverage_type: type,
      core_contents: [arg + ' — completare con argomenti specifici del modulo'],
      key_concepts:  [arg],
      _freqs:        freqsByCdl
    };
  });

  var order = {universal:0, profile_dependent:1, discretionary:2};
  modulesData.sort(function(a,b){ return order[a.coverage_type]-order[b.coverage_type]; });

  var programProfiles = cdls.map(function(cdl){
    var n          = counts[cdl]||0;
    var priModules = modulesData
      .filter(function(m){ return m.coverage_type==='universal'||(m._freqs[cdl]||0)>=SETTINGS.threshProfile; })
      .map(function(m){ return m.name; });
    var priText = priModules.slice(0,4).join(', ') + (priModules.length>4?'...':'');
    return {
      name: cdl,
      priority_modules: '[N=' + n + ' programmi] Focus su: ' + priText + '. Completare con AI.'
    };
  });

  CURRENT_PROJECT.framework = {
    id:               materia + '_framework',
    name:             CURRENT_PROJECT.materiaName || materia,
    subject:          materia,
    context_strategy: '',
    _meta:            { ssd: CURRENT_PROJECT.materiaSsd||'', cdls: cdls, counts: counts },
    content: {
      target_degrees:      cdls,
      evaluation_criteria: {
        contenuti_core:           'Quali argomenti fondamentali sono presenti/assenti?',
        approccio_metodologico:   'Approccio teorico/descrittivo vs empirico/applicativo?',
        orientamento_disciplinare:'Orientamento prevalente e scuola di pensiero?'
      },
      program_profiles: programProfiles,
      syllabus_modules: modulesData
    }
  };

  document.getElementById('exportContent').style.display = 'block';
  document.getElementById('exportEmpty').style.display   = 'none';
  autoSaveProject();
  showView('builderView');
  showToast('✅ Framework generato! Clicca "Genera con AI" per completarlo.');
}

async function generateWithAI() {
  if (!SETTINGS.apiKey) {
    showToast('Inserisci la chiave API nelle impostazioni', 'warning');
    openModal('settingsModal');
    return;
  }
  if (!CURRENT_PROJECT || !CURRENT_PROJECT.framework) {
    showToast('Genera prima il framework dalla tabella frequenze', 'warning');
    return;
  }

  var btn   = document.getElementById('btnGenerateAI');
  var label = document.getElementById('aiGeneratingLabel');
  btn.disabled        = true;
  label.style.display = 'inline';

  try {
    var fw     = CURRENT_PROJECT.framework;
    var cdls   = fw._meta ? fw._meta.cdls   : fw.content.target_degrees;
    var counts = fw._meta ? fw._meta.counts : {};
    var modules = fw.content.syllabus_modules;

    // ── Sommario frequenze ──────────────────────
    var freqSummary = '';
    modules.forEach(function(m) {
      var fl = cdls.map(function(c) {
        return c + ':' + (m._freqs ? (m._freqs[c] !== undefined ? m._freqs[c] : '?') : '?') + '%';
      }).join(', ');
      freqSummary += '- ' + m.name + ' [' + m.coverage_type + '] -> ' + fl + '\n';
    });

    // ── Top-6 argomenti discriminanti tra CdL ──
    var discriminanti = [];
    modules.forEach(function(m) {
      if (m._freqs && cdls.length >= 2) {
        var vals  = cdls.map(function(c) { return m._freqs[c] || 0; });
        var delta = Math.max.apply(null, vals) - Math.min.apply(null, vals);
        discriminanti.push({ name: m.name, delta: delta, vals: vals });
      }
    });
    discriminanti.sort(function(a, b) { return b.delta - a.delta; });
    discriminanti = discriminanti.slice(0, 6);
    var discText = '';
    discriminanti.forEach(function(d) {
      var vstr = cdls.map(function(c, i) { return c + '=' + d.vals[i] + '%'; }).join(', ');
      discText += '- "' + d.name + '": delta ' + d.delta + '% (' + vstr + ')\n';
    });

    var cdlSample = '';
    cdls.forEach(function(c) {
      cdlSample += '- ' + c + ': ' + (counts[c] || '?') + ' programmi\n';
    });
    var totale = cdls.reduce(function(s, c) { return s + (counts[c] || 0); }, 0);

    // ════════════════════════════
    //  STEP 1 — Normalizzazione
    // ════════════════════════════
    showToast('Step 1/2 - Normalizzazione argomenti...', 'warning');

    var argsText = '';
    modules.forEach(function(m, i) {
      var fs = cdls.map(function(c) {
        return c + ':' + (m._freqs ? (m._freqs[c] || 0) : 0) + '%';
      }).join(', ');
      argsText += (i + 1) + '. "' + m.name + '" [' + m.coverage_type + '] - ' + fs + '\n';
    });

    var step1System =
      'Sei un esperto di didattica universitaria italiana e di analisi curricolare. '
      + 'Ricevi una lista di argomenti estratti automaticamente da programmi universitari italiani tramite OCR e AI. '
      + 'Produci una lista normalizzata e pulita. '
      + 'Rispondi SOLO con un oggetto JSON valido, senza markdown, senza spiegazioni.';

    var step1User =
      'Materia: ' + fw.name + ' (SSD: ' + (CURRENT_PROJECT.materiaSsd || '') + ')\n'
      + 'Classi di laurea:\n' + cdlSample + '\n'
      + 'ARGOMENTI GREZZI:\n' + argsText + '\n'
      + 'ISTRUZIONI:\n'
      + '1. Fondi duplicati semantici\n'
      + '2. Elimina argomenti <5% su TUTTI i CdL\n'
      + '3. Mantieni argomenti >=5% su almeno un CdL\n'
      + '4. Nomi in italiano accademico standard\n'
      + '5. coverage_type: "universal">=65% tutti, "profile_dependent">=50% almeno uno, "discretionary" altrimenti\n'
      + '6. Indica fused_from per fusioni\n\n'
      + 'JSON atteso:\n'
      + '{"normalized_modules":[{"name":"...","coverage_type":"...","freqs":{"' + cdls[0] + '":75},'
      + '"fused_from":["..."],"key_concepts":["..."]}],'
      + '"removed":["..."],"normalization_notes":"..."}';

    var r1 = await fetch('https://api.openai.com/v1/chat/completions', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json', 'Authorization': 'Bearer ' + SETTINGS.apiKey },
      body: JSON.stringify({
        model: SETTINGS.model,
        messages: [
          { role: 'system', content: step1System },
          { role: 'user', content: step1User }
        ],
        temperature: 0.1, max_tokens: 2500
      })
    });
    if (!r1.ok) throw new Error('API Step1 error: ' + r1.status);
    var d1   = await r1.json();
    var raw1 = d1.choices[0].message.content.trim();
    var step1Result;
    try {
      var m1 = raw1.match(/\{[\s\S]*\}/);
      step1Result = JSON.parse(m1 ? m1[0] : raw1);
    } catch(e) {
      throw new Error('Step1 risposta non valida: ' + raw1.substring(0, 200));
    }

    var normModules = step1Result.normalized_modules || [];
    if (!normModules.length) throw new Error('Step1 non ha prodotto moduli');

    var ordMap = { universal: 0, profile_dependent: 1, discretionary: 2 };
    CURRENT_PROJECT.framework.content.syllabus_modules = normModules.map(function(m) {
      var cc = (m.fused_from && m.fused_from.length > 1)
        ? ['Comprende: ' + m.fused_from.join(', ')]
        : [m.name + ' - completare con i contenuti specifici'];
      return {
        name: m.name,
        coverage_type: m.coverage_type,
        core_contents: cc,
        key_concepts: m.key_concepts || [m.name],
        _freqs: m.freqs || {}
      };
    });
    CURRENT_PROJECT.framework.content.syllabus_modules.sort(function(a, b) {
      return ordMap[a.coverage_type] - ordMap[b.coverage_type];
    });
    renderModulesList();
    showToast('Step 1 OK — ' + normModules.length + ' moduli normalizzati');

    // ════════════════════════════
    //  STEP 2 — Testi e criteri
    // ════════════════════════════
    showToast('Step 2/2 - Generazione testi...', 'warning');

    var modNorm = CURRENT_PROJECT.framework.content.syllabus_modules;
    var univStr = modNorm.filter(function(m) { return m.coverage_type === 'universal'; })
                         .map(function(m) { return m.name; }).join(', ') || 'nessuno';
    var profStr = modNorm.filter(function(m) { return m.coverage_type === 'profile_dependent'; })
                         .map(function(m) { return m.name; }).join(', ') || 'nessuno';
    var discStr = modNorm.filter(function(m) { return m.coverage_type === 'discretionary'; })
                         .map(function(m) { return m.name; }).join(', ') || 'nessuno';

    var modText = '';
    modNorm.forEach(function(m) {
      var fs = cdls.map(function(c) {
        return c + ':' + (m._freqs ? (m._freqs[c] || 0) : 0) + '%';
      }).join(', ');
      modText += '- "' + m.name + '" [' + m.coverage_type + '] ' + fs + '\n';
    });

    // Template moduli da arricchire
    var modEnrTpl = modNorm.map(function(m, i) {
      return '    {"name":"' + m.name + '","core_contents":"...","key_concepts":["...","...","..."]}' 
             + (i < modNorm.length - 1 ? ',' : '');
    }).join('\n');

    // Template profili — esempio esplicito multi-riga per guidare il modello
    var profExamples = cdls.map(function(c) {
      var topMods = modNorm
        .filter(function(m) { return (m._freqs && m._freqs[c]) ? m._freqs[c] >= 50 : false; })
        .sort(function(a, b) { return (b._freqs[c] || 0) - (a._freqs[c] || 0); })
        .slice(0, 3)
        .map(function(m) { return m.name + ' (' + (m._freqs[c] || 0) + '%)'; })
        .join(', ');
      var lowMods = modNorm
        .filter(function(m) { return (m._freqs && m._freqs[c]) ? m._freqs[c] < 40 : true; })
        .slice(0, 2)
        .map(function(m) { return m.name; })
        .join(', ');
      return '    {"name":"' + c + '","priority_modules":"[ESEMPIO — sostituire] '
        + 'Moduli prioritari: ' + (topMods || 'da determinare') + '. '
        + 'Approccio didattico consigliato: empirico-sperimentale. '
        + 'Meno rilevanti rispetto agli altri CdL: ' + (lowMods || 'nessuno') + '."}';
    }).join(',\n');

    // ── Costruzione chiavi criteri da moduli reali ──
    // Estraiamo i 5 moduli universal più frequenti come base delle chiavi diagnostiche
    var topUniversal = modNorm
      .filter(function(m) { return m.coverage_type === 'universal'; })
      .slice(0, 5);
    var topProfDependent = modNorm
      .filter(function(m) { return m.coverage_type === 'profile_dependent'; })
      .slice(0, 3);
    var criteriaExamples = topUniversal.concat(topProfDependent).slice(0, 5);
    var criteriaKeysExample = criteriaExamples.map(function(m) {
      // genera slug snake_case dal nome del modulo
      var slug = m.name
        .toLowerCase()
        .replace(/[àáâã]/g, 'a').replace(/[èéêë]/g, 'e')
        .replace(/[ìíîï]/g, 'i').replace(/[òóôõ]/g, 'o')
        .replace(/[ùúûü]/g, 'u')
        .replace(/[^a-z0-9\s]/g, '')
        .trim()
        .replace(/\s+/g, '_')
        .substring(0, 40);
      var freq = cdls.map(function(c) {
        return c + '=' + (m._freqs ? (m._freqs[c] || 0) : '?') + '%';
      }).join(', ');
      return '    "' + slug + '": "Il programma distingue/approfondisce [aspetto specifico di '
        + m.name + ']? [freq: ' + freq + ']"';
    }).join(',\n');

    var step2System =
      'Sei un consulente editoriale specializzato in testi universitari italiani. '
      + 'Produci testi operativi per un framework di analisi curricolare usato da promotori editoriali Zanichelli. '
      + 'Scrivi in italiano professionale. Sii specifico, basati sui dati reali di frequenza. '
      + 'Rispondi SOLO con un oggetto JSON valido, senza markdown, senza testo prima o dopo le parentesi graffe.';

    var step2User =
      'Framework: "' + fw.name + '" (SSD: ' + (CURRENT_PROJECT.materiaSsd || '') + ')\n'
      + 'Campione: ' + cdls.map(function(c) { return c + ' n=' + (counts[c] || '?'); }).join(', ')
      + ' — totale ' + totale + ' programmi\n\n'
      + 'MODULI NORMALIZZATI:\n'
      + 'Universal: ' + univStr + '\n'
      + 'Profile-dependent: ' + profStr + '\n'
      + 'Discretionary: ' + discStr + '\n\n'
      + 'FREQUENZE DETTAGLIATE:\n' + modText + '\n'
      + 'ARGOMENTI PIÙ DISCRIMINANTI (maggiore delta tra CdL):\n' + discText + '\n'
      + 'NOTE NORMALIZZAZIONE: ' + (step1Result.normalization_notes || '') + '\n\n'

      // ── REGOLE DETTAGLIATE ──
      + '=== REGOLE OBBLIGATORIE ===\n\n'

      + '1) context_strategy: MAX 2 righe. Indica il discriminante principale tra '
      + cdls.join(' e ') + ' e cosa cerca un promotore editoriale per distinguere i testi adatti.\n\n'

      + '2) modules_enriched: per OGNI modulo scrivi core_contents come testo narrativo di 2-3 '
      + 'righe con teorie specifiche, autori rilevanti, approcci metodologici. '
      + 'key_concepts: array di 3-5 termini tecnici precisi.\n\n'

      + '3) evaluation_criteria: ESATTAMENTE 5 voci. '
      + 'Le CHIAVI devono essere snake_case descrittive del contenuto disciplinare reale, '
      + 'DERIVATE dai nomi dei moduli Universal e Profile-dependent (NON generiche come '
      + '"criterio_1", "chiave_A", "slug_argomento"). '
      + 'I VALORI devono essere domande diagnostiche verificabili, specifiche per ' + fw.name + '. '
      + 'Esempi di CHIAVI CORRETTE per questa materia: '
      + criteriaExamples.slice(0, 3).map(function(m) {
          return '"' + m.name.toLowerCase().replace(/\s+/g, '_').substring(0, 30) + '"';
        }).join(', ') + '. '
      + 'Esempi di CHIAVI SBAGLIATE (da non usare mai): '
      + '"chiave_descrittiva_A", "slug_argomento_1", "criterio_generico", "NOTA".\n\n'

      + '4) program_profiles: per OGNI CdL scrivi un testo di 4-5 righe operative. '
      + 'Il testo DEVE contenere obbligatoriamente: '
      + '(a) i 3-4 moduli prioritari con la loro percentuale di frequenza es. "Memoria a lungo termine (85%)", '
      + '(b) l\'approccio didattico da privilegiare nella scelta del testo, '
      + '(c) i moduli meno rilevanti rispetto agli altri CdL con percentuale, '
      + '(d) un consiglio operativo per il promotore. '
      + 'Esempio di testo CORRETTO per un CdL: "Cruciali Basi biologiche del comportamento (88%) '
      + 'e Memoria a lungo termine (81%). Privilegiare testi con forte ancoraggio neuroscentifico '
      + 'e capitoli sperimentali. Meno rilevanti rispetto a LM-51: Storia della psicologia (32%) '
      + 'e Psicopatologia (28%). Consiglio: verificare presenza di laboratori e casi clinici."\n\n'

      // ── TEMPLATE JSON ──
      + '=== JSON DA PRODURRE (rispetta ESATTAMENTE questa struttura) ===\n'
      + '{\n'
      + '  "context_strategy": "testo operativo max 2 righe",\n'
      + '  "modules_enriched": [\n' + modEnrTpl + '\n  ],\n'
      + '  "evaluation_criteria": {\n'
      + criteriaKeysExample + '\n'
      + '  },\n'
      + '  "program_profiles": [\n' + profExamples + '\n  ]\n'
      + '}';

    var r2 = await fetch('https://api.openai.com/v1/chat/completions', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json', 'Authorization': 'Bearer ' + SETTINGS.apiKey },
      body: JSON.stringify({
        model: SETTINGS.model,
        messages: [
          { role: 'system', content: step2System },
          { role: 'user', content: step2User }
        ],
        temperature: 0.3,
        max_tokens: 3500    // aumentato per profili più lunghi
      })
    });
    if (!r2.ok) throw new Error('API Step2 error: ' + r2.status);
    var d2   = await r2.json();
    var raw2 = d2.choices[0].message.content.trim();
    var step2Result;
    try {
      var m2 = raw2.match(/\{[\s\S]*\}/);
      step2Result = JSON.parse(m2 ? m2[0] : raw2);
    } catch(e) {
      throw new Error('Step2 risposta non valida: ' + raw2.substring(0, 200));
    }

    // ── Applica context_strategy ──────────────
    if (step2Result.context_strategy) {
      CURRENT_PROJECT.framework.context_strategy = step2Result.context_strategy;
      var csEl = document.getElementById('fwContextStrategy');
      if (csEl) csEl.value = step2Result.context_strategy;
    }

    // ── Applica moduli arricchiti ─────────────
    if (step2Result.modules_enriched && step2Result.modules_enriched.length) {
      step2Result.modules_enriched.forEach(function(en, i) {
        if (CURRENT_PROJECT.framework.content.syllabus_modules[i]) {
          if (en.core_contents) {
            CURRENT_PROJECT.framework.content.syllabus_modules[i].core_contents = [en.core_contents];
          }
          if (en.key_concepts && en.key_concepts.length) {
            CURRENT_PROJECT.framework.content.syllabus_modules[i].key_concepts = en.key_concepts;
          }
        }
      });
      renderModulesList();
    }

    // ── Applica evaluation_criteria (filtro robusto) ──
    if (step2Result.evaluation_criteria && typeof step2Result.evaluation_criteria === 'object') {
      var cleaned = {};

      // Pattern di chiavi generiche/template da scartare (lista exhaustiva)
      var badPatterns = [
        /^chiave_/i,
        /^slug_/i,
        /^criterio_/i,
        /^argomento_/i,
        /^campo_/i,
        /^esempio_/i,
        /^key_/i,
        /^criterion_/i,
        /^_/,                        // chiavi che iniziano con _
        /^(NOTA|nota|note|NOTE)$/,
        /^[a-z]$/,                   // chiave di un solo carattere
        /^[A-Z]$/,
        /\[SOSTITUIRE\]/i,
        /\[esempio\]/i
      ];

      var isBadKey = function(k) {
        if (!k || typeof k !== 'string') return true;
        return badPatterns.some(function(p) { return p.test(k); });
      };

      Object.entries(step2Result.evaluation_criteria).forEach(function(entry) {
        var k = entry[0];
        var v = entry[1];
        if (
          !isBadKey(k)
          && typeof v === 'string'
          && v.trim().length > 8          // valore non vuoto né placeholder
          && !v.includes('[SOSTITUIRE')
          && !v.includes('[esempio')
        ) {
          cleaned[k] = v.trim();
        }
      });

      // Se il filtro ha eliminato tutto (modello refrattario),
      // rigenera le chiavi dai nomi dei moduli in modo deterministico
      if (Object.keys(cleaned).length < 3) {
        criteriaExamples.slice(0, 5).forEach(function(m) {
          var slug = m.name
            .toLowerCase()
            .replace(/[àáâã]/g, 'a').replace(/[èéêë]/g, 'e')
            .replace(/[ìíîï]/g, 'i').replace(/[òóôõ]/g, 'o')
            .replace(/[ùúûü]/g, 'u')
            .replace(/[^a-z0-9\s]/g, '')
            .trim().replace(/\s+/g, '_').substring(0, 40);
          var freq = cdls.map(function(c) {
            return c + '=' + (m._freqs ? (m._freqs[c] || 0) : '?') + '%';
          }).join(', ');
          // Usa il valore della AI se la chiave corrisponde, altrimenti placeholder
          var aiValue = step2Result.evaluation_criteria[slug]
            || 'Il programma approfondisce ' + m.name + '? [' + freq + ']';
          cleaned[slug] = aiValue;
        });
      }

      CURRENT_PROJECT.framework.content.evaluation_criteria = cleaned;
      renderEvaluationCriteria();
    }

    // ── Applica program_profiles ──────────────
    if (step2Result.program_profiles && step2Result.program_profiles.length) {
      step2Result.program_profiles.forEach(function(pp, i) {
        if (CURRENT_PROJECT.framework.content.program_profiles[i]) {
          if (pp.priority_modules) {
            CURRENT_PROJECT.framework.content.program_profiles[i].priority_modules = pp.priority_modules;
          }
          if (pp.name) {
            CURRENT_PROJECT.framework.content.program_profiles[i].name = pp.name;
          }
        }
      });
      renderProfilesList();
    }

    refreshJsonPreview();
    autoSaveProject();
    showToast('✅ Framework completato — pronto per export!');

  } catch(e) {
    console.error('generateWithAI error:', e);
    showToast('❌ Errore: ' + e.message, 'error');
  } finally {
    btn.disabled        = false;
    label.style.display = 'none';
  }
}

function getFinalJson() {
  if (!CURRENT_PROJECT || !CURRENT_PROJECT.framework) return null;
  var fw = CURRENT_PROJECT.framework;

  var readField = function(id) {
    var el = document.getElementById(id);
    return (el && el.value.trim()) ? el.value.trim() : null;
  };

  var id               = readField('fwId')             || fw.id;
  var name             = readField('fwName')            || fw.name;
  var subjectRaw       = readField('fwSubject')         || fw.subject || '';
  var context_strategy = readField('fwContextStrategy') || fw.context_strategy || '';

  var subjectMap = {
    'psicologia generale':'psicologia','psicologia clinica':'psicologia',
    'psic. dello sviluppo':'psicologia','psicologia dello sviluppo':'psicologia',
    'psicologia sociale':'psicologia','psicometria':'psicologia',
    'antropologia culturale':'scienze_umane','sociologia generale':'scienze_umane',
    'chimica generale':'chimica','chimica organica':'chimica',
    'fisica':'fisica','fisica generale':'fisica',
    'matematica':'matematica','analisi matematica':'matematica',
    'microeconomia':'economia','macroeconomia':'economia','economia politica':'economia',
    'biologia':'biologia','biologia cellulare':'biologia','biologia molecolare':'biologia',
    'biochimica':'biologia','genetica':'biologia','microbiologia':'biologia','istologia':'istologia'
  };
  var subjectNormalized = subjectMap[subjectRaw.toLowerCase().trim()] || subjectRaw;

  var syllabus_modules = fw.content.syllabus_modules.map(function(m){
    var core_str = Array.isArray(m.core_contents)
      ? m.core_contents.filter(Boolean).join('. ')
      : (m.core_contents||'');
    var kc = (m.key_concepts||[]).map(function(k){
      return typeof k==='object'?(k.concetto||k.name||JSON.stringify(k)):String(k);
    }).filter(Boolean);
    return {name:m.name, core_contents:core_str, key_concepts:kc};
  });

  var evaluation_criteria = Object.assign({}, fw.content.evaluation_criteria||{});
  delete evaluation_criteria._note;

  return {
    id:id, name:name, subject:subjectNormalized, context_strategy:context_strategy,
    content:{
      target_degrees:   fw.content.target_degrees,
      evaluation_criteria: evaluation_criteria,
      program_profiles: fw.content.program_profiles.map(function(p){
        return {name:p.name, priority_modules:p.priority_modules};
      }),
      syllabus_modules: syllabus_modules
    }
  };
}

function downloadCsv() {
  if (!CURRENT_PROJECT) return;
  var materia = CURRENT_PROJECT.materiaId;
  if (!materia || !CURRENT_PROJECT.allArgs[materia]) {
    showToast('⚠️ Nessun dato frequenze da esportare','warning'); return;
  }
  var cdlData = CURRENT_PROJECT.allArgs[materia];
  var counts  = CURRENT_PROJECT.sampleCounts[materia]||{};
  var cdls    = Object.keys(cdlData);
  var allArgs = [...new Set(cdls.flatMap(function(c){ return Object.keys(cdlData[c]); }))].sort();
  var header  = ['Argomento','Tipo',...cdls.map(function(c){ return c+' (n='+(counts[c]||0)+') %'; })].join(',');
  var rows = allArgs.map(function(arg){
    var freqsByCdl={};
    cdls.forEach(function(cdl){
      var n=counts[cdl]||1;
      freqsByCdl[cdl]=Math.round(((cdlData[cdl][arg]||0)/n)*100);
    });
    var type=getArgType(arg,freqsByCdl,cdls,counts);
    return ['"'+arg+'"',type,...cdls.map(function(c){ return freqsByCdl[c]; })].join(',');
  });
  var csv  = [header,...rows].join('\n');
  var blob = new Blob([csv],{type:'text/csv;charset=utf-8;'});
  var a    = document.createElement('a');
  a.href   = URL.createObjectURL(blob);
  a.download = (CURRENT_PROJECT.name||materia).replace(/\s/g,'_')+'_frequenze.csv';
  document.body.appendChild(a); a.click();
  document.body.removeChild(a); URL.revokeObjectURL(a.href);
  showToast('✅ CSV scaricato');
}

    // Recupera testo da usare
    var activeTab = document.getElementById('tabPdf').classList.contains('active')
      ? 'pdf' : 'text';
    var inputText = activeTab === 'pdf'
      ? BOOK_PDF_TEXT
      : document.getElementById('bookTextInput').value.trim();

    if (!inputText || inputText.length < 50) {
      showToast('⚠️ Nessun testo disponibile. Carica un PDF o incolla il testo.', 'warning');
      return;
    }
    if (!SETTINGS.apiKey) {
      showToast('⚠️ Inserisci la chiave API nelle impostazioni', 'warning');
      openModal('settingsModal'); return;
    }
    await generateBookJson(inputText);
  }
}

// ── Switch tab PDF / Testo ───────────────────
function switchBookTab(tab) {
  document.getElementById('bookTabPdf').style.display  = tab === 'pdf'  ? 'block' : 'none';
  document.getElementById('bookTabText').style.display = tab === 'text' ? 'block' : 'none';
  document.getElementById('tabPdf').classList.toggle('active',  tab === 'pdf');
  document.getElementById('tabText').classList.toggle('active', tab === 'text');
}

// ── Gestione PDF ─────────────────────────────
function setupBookDropZone() {
  var dz = document.getElementById('bookDropZone');
  if (!dz) return;
  dz.addEventListener('dragover', function(e) {
    e.preventDefault(); dz.classList.add('drag-over');
  });
  dz.addEventListener('dragleave', function() {
    dz.classList.remove('drag-over');
  });
  dz.addEventListener('drop', function(e) {
    e.preventDefault();
    dz.classList.remove('drag-over');
    var f = e.dataTransfer.files[0];
    if (f && f.name.endsWith('.pdf')) handleBookPdf(f);
    else showToast('⚠️ Carica un file PDF', 'warning');
  });
}

async function handleBookPdf(file) {
  if (!file) return;
  var statusEl = document.getElementById('bookPdfStatus');
  statusEl.innerHTML =
    '<div class="file-item processing">'
    + '<div class="file-icon">⚙️</div>'
    + '<div class="file-info"><div class="file-name">' + escHtml(file.name) + '</div>'
    + '<div class="file-meta">Estrazione testo in corso...</div></div>'
    + '</div>';

  try {
    var text = await extractTextFromPdf(file);

    // Verifica se il testo è sufficiente
    if (!text || text.length < 100) {
      // PDF non leggibile — suggerisci il tab testo
      statusEl.innerHTML =
        '<div class="file-item error">'
        + '<div class="file-icon">❌</div>'
        + '<div class="file-info">'
        + '<div class="file-name">' + escHtml(file.name) + '</div>'
        + '<div class="file-meta">PDF non leggibile automaticamente.</div>'
        + '</div></div>';
      var callout = '<div class="callout warning" style="margin-top:10px">'
        + '<span class="ci">⚠️</span>'
        + '<div>Questo PDF è scansionato o protetto. '
        + 'Aprilo in Word o Acrobat, copia il testo dell\'indice e usa il tab '
        + '<strong>"Incolla testo"</strong>.</div></div>';
      statusEl.innerHTML += callout;
      BOOK_PDF_TEXT = '';
      return;
    }

    // Testo estratto correttamente
    BOOK_PDF_TEXT = text;
    statusEl.innerHTML =
      '<div class="file-item done">'
      + '<div class="file-icon">✅</div>'
      + '<div class="file-info">'
      + '<div class="file-name">' + escHtml(file.name) + '</div>'
      + '<div class="file-meta">Testo estratto: '
      + text.length + ' caratteri — pronto per la generazione.</div>'
      + '</div></div>';

  } catch(e) {
    statusEl.innerHTML =
      '<div class="file-item error">'
      + '<div class="file-icon">❌</div>'
      + '<div class="file-info"><div class="file-name">' + escHtml(file.name) + '</div>'
      + '<div class="file-meta">Errore: ' + e.message + '</div>'
      + '</div></div>';
    BOOK_PDF_TEXT = '';
  }
}

// ── Generazione JSON con AI ──────────────────
async function generateBookJson(inputText) {
  var btn = document.getElementById('btnBookNext');
  btn.disabled = true;
  document.getElementById('bookProgress').style.display = 'block';

  function setProgress(pct, label) {
    document.getElementById('bookProgressLabel').textContent = label;
    document.getElementById('bookProgressPct').textContent   = pct + '%';
    document.getElementById('bookProgressFill').style.width  = pct + '%';
  }

  try {
    setProgress(20, 'Analisi dell\'indice in corso...');

    var meta = {
      id:        document.getElementById('bookId').value.trim(),
      title:     document.getElementById('bookTitle').value.trim(),
      author:    document.getElementById('bookAuthor').value.trim(),
      publisher: document.getElementById('bookPublisher').value.trim(),
      subject:   document.getElementById('bookSubject').value.trim(),
      type:      document.getElementById('bookType').value
    };

    var systemPrompt =
      'Sei un esperto di editoria universitaria italiana. '
      + 'Ricevi il testo grezzo di un indice di un manuale universitario, '
      + 'che potrebbe essere mal formattato o disordinato a causa dell\'estrazione da PDF. '
      + 'Il tuo compito è ricostruire la struttura gerarchica esatta: '
      + 'capitoli e relative sezioni/paragrafi con i numeri di pagina. '
      + 'Rispondi SOLO con un oggetto JSON valido, senza markdown, senza spiegazioni.';

    var userPrompt =
      'Metadati del volume:\n'
      + '- Titolo: ' + meta.title + '\n'
      + '- Autore: ' + meta.author + '\n'
      + '- Materia: ' + meta.subject + '\n\n'
      + 'TESTO GREZZO DELL\'INDICE:\n'
      + inputText.substring(0, 8000) + '\n\n'
      + 'ISTRUZIONI:\n'
      + '1. Ricostruisci la struttura gerarchica ignorando artefatti di formattazione\n'
      + '2. Ogni capitolo ha: number (intero), title (stringa), page_start (intero), sections (array)\n'
      + '3. Ogni sezione ha: number (stringa es. "1.1"), title (stringa), page_start (intero)\n'
      + '4. Se il numero di pagina non è leggibile usa 0\n'
      + '5. Ignora prefazioni, bibliografie, indici analitici — solo i capitoli principali\n'
      + '6. Normalizza i titoli in italiano corretto\n\n'
      + 'JSON atteso:\n'
      + '{\n'
      + '  "id": "' + meta.id + '",\n'
      + '  "title": "' + meta.title + '",\n'
      + '  "author": "' + meta.author + '",\n'
      + '  "publisher": "' + meta.publisher + '",\n'
      + '  "subject": "' + meta.subject + '",\n'
      + '  "type": "' + meta.type + '",\n'
      + '  "chapters": [\n'
      + '    {\n'
      + '      "number": 1,\n'
      + '      "title": "...",\n'
      + '      "page_start": 1,\n'
      + '      "sections": [\n'
      + '        {"number": "1.1", "title": "...", "page_start": 1}\n'
      + '      ]\n'
      + '    }\n'
      + '  ]\n'
      + '}';

    setProgress(50, 'Chiamata AI in corso...');

    var resp = await fetch('https://api.openai.com/v1/chat/completions', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': 'Bearer ' + SETTINGS.apiKey
      },
      body: JSON.stringify({
        model: SETTINGS.model,
        messages: [
          { role: 'system', content: systemPrompt },
          { role: 'user',   content: userPrompt }
        ],
        temperature: 0.1,
        max_tokens: 4000
      })
    });

    if (!resp.ok) throw new Error('API error: ' + resp.status);
    var data = await resp.json();
    var raw  = data.choices[0].message.content.trim();

    setProgress(80, 'Parsing JSON...');

    var parsed;
    try {
      var match = raw.match(/\{[\s\S]*\}/);
      parsed = JSON.parse(match ? match[0] : raw);
    } catch(e) {
      throw new Error('Risposta AI non valida: ' + raw.substring(0, 200));
    }

    // Assicura che i metadati siano quelli inseriti dall'utente
    parsed.id        = meta.id;
    parsed.title     = meta.title;
    parsed.author    = meta.author;
    parsed.publisher = meta.publisher;
    parsed.subject   = meta.subject;
    parsed.type      = meta.type;

    BOOK_JSON = parsed;

    setProgress(100, 'Completato!');
    document.getElementById('bookJsonPreview').textContent =
      JSON.stringify(BOOK_JSON, null, 2);

    BOOK_STEP = 3;
    updateBookUI();
    showToast('✅ JSON generato — ' + (parsed.chapters || []).length + ' capitoli');

  } catch(e) {
    console.error('generateBookJson error:', e);
    showToast('❌ Errore: ' + e.message, 'error');
    document.getElementById('bookProgress').style.display = 'none';
  } finally {
    btn.disabled = false;
  }
}

// ── Download e copia ─────────────────────────
function downloadBookJson() {
  if (!BOOK_JSON) return;
  var blob = new Blob(
    [JSON.stringify(BOOK_JSON, null, 2)],
    { type: 'application/json' }
  );
  var a = document.createElement('a');
  a.href = URL.createObjectURL(blob);
  a.download = BOOK_JSON.id + '.json';
  document.body.appendChild(a);
  a.click();
  document.body.removeChild(a);
  URL.revokeObjectURL(a.href);
  showToast('✅ JSON scaricato: ' + BOOK_JSON.id + '.json');
}

function copyBookJson() {
  if (!BOOK_JSON) return;
  navigator.clipboard.writeText(JSON.stringify(BOOK_JSON, null, 2))
    .then(function() { showToast('✅ JSON copiato negli appunti'); })
    .catch(function() { showToast('❌ Errore copia', 'error'); });
}
