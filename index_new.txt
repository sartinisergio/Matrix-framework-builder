function loadSettings() {
  try {
    var s = JSON.parse(localStorage.getItem('mfb_settings') || '{}');
    if (s.apiKey)                SETTINGS.apiKey                = s.apiKey;
    if (s.model)                 SETTINGS.model                 = s.model;
    if (s.threshUniversal)       SETTINGS.threshUniversal       = +s.threshUniversal;
    if (s.threshProfile)         SETTINGS.threshProfile         = +s.threshProfile;
    if (s.smallSampleCorrection) SETTINGS.smallSampleCorrection = s.smallSampleCorrection;
  } catch(e) { console.warn('loadSettings error:', e); }
}

function saveSettings() {
  SETTINGS.apiKey                = document.getElementById('settingsApiKey').value.trim();
  SETTINGS.model                 = document.getElementById('settingsModel').value;
  SETTINGS.threshUniversal       = +document.getElementById('threshUniversal').value;
  SETTINGS.threshProfile         = +document.getElementById('threshProfile').value;

  localStorage.setItem('mfb_settings', JSON.stringify(SETTINGS));

}

// ════════════════════════════════════════════════
function getAllProjects() {
  var projects = [];
  for (var i = 0; i < localStorage.length; i++) {
    var key = localStorage.key(i);
    if (key && key.startsWith('mfb_project_')) {
      try {
        var p = JSON.parse(localStorage.getItem(key));
        if (p && p.id) projects.push(p);
      } catch(e) {}
    }
  }
  projects.sort(function(a,b){ return (b.updatedAt||0) - (a.updatedAt||0); });
  return projects;
}

function saveProject(project) {
  project.updatedAt = Date.now();
  localStorage.setItem('mfb_project_' + project.id, JSON.stringify(project));
}

function deleteProject(projectId) {
  localStorage.removeItem('mfb_project_' + projectId);
}

function createNewProject(name, materiaId, materiaName, materiaSsd, cdls) {
  var id = 'proj_' + Date.now();
  var project = {
    id:          id,
    name:        name,
    materiaId:   materiaId,
    materiaName: materiaName,
    materiaSsd:  materiaSsd,
    cdls:        cdls,
    createdAt:   Date.now(),
    updatedAt:   Date.now(),
    status:      'new',    // new | working | framework | exported
    files:       [],
    allArgs:     {},
    sampleCounts:{},
    framework:   null
  };
  saveProject(project);
  return project;
}

function getProjectStatus(project) {
  if (project.framework) return 'exported';
  if (project.allArgs && Object.keys(project.allArgs).length) return 'framework';
  if (project.files && project.files.filter(function(f){ return f.status==='done'; }).length)
    return 'working';
  return 'new';
}

function toggleWizardCdl(id) {
  var idx = wizardSelectedCdls.indexOf(id);
  if (idx >= 0) {
    wizardSelectedCdls.splice(idx, 1);
  } else {
    wizardSelectedCdls.push(id);
  }
  var btn = document.getElementById('wcdl-' + id.replace(/\//g,'_'));
  if (btn) btn.classList.toggle('selected', wizardSelectedCdls.indexOf(id) >= 0);
  renderWizardCdlSelected();
}

  wizardSelectedCdls.push(val);
  document.getElementById('customCdlInput').value = '';
  renderWizardCdlSelected();
}

function renderWizardCdlSelected() {
  var el = document.getElementById('wizardCdlSelected');

  el.innerHTML = '<div style="margin-top:8px;font-size:12px;color:var(--text-2)">'
    + '<strong>Selezionati:</strong> '

function updateWizardUI() {
  [1,2,3].forEach(function(i) {
    var step  = document.getElementById('wstep'  + i);
    var panel = document.getElementById('wpanel' + i);
    step.classList.toggle('active', i === wizardStep);
    step.classList.toggle('done',   i <  wizardStep);
    panel.classList.toggle('active', i === wizardStep);
  });
  var btnBack = document.getElementById('btnWizardBack');
  var btnNext = document.getElementById('btnWizardNext');
  btnBack.style.display = wizardStep > 1 ? 'inline-flex' : 'none';
  btnNext.innerHTML = wizardStep < 3
    ? 'Avanti <i class="fas fa-arrow-right"></i>'
    : '<i class="fas fa-check"></i> Crea progetto';
}

function wizardNext() {
  if (wizardStep === 1) {
    var name = document.getElementById('newProjectName').value.trim();
    if (!name) { showToast('⚠️ Inserisci un nome per il progetto', 'warning'); return; }
    wizardStep = 2;
    updateWizardUI();
  } else if (wizardStep === 2) {
    if (!wizardSelectedMateria) {
      showToast('⚠️ Seleziona una materia', 'warning'); return;
    }
    if (wizardSelectedMateria === 'custom') {
      var cname = document.getElementById('customMateriaName').value.trim();
      if (!cname) { showToast('⚠️ Inserisci il nome della materia', 'warning'); return; }
    }
    wizardStep = 3;
    updateWizardUI();
  } else if (wizardStep === 3) {
    if (!wizardSelectedCdls.length) {
      showToast('⚠️ Seleziona almeno una classe di laurea', 'warning'); return;
    }
    // Crea il progetto
    var pname    = document.getElementById('newProjectName').value.trim();
    var mId      = wizardSelectedMateria;
    var mInfo    = MATERIE.find(function(m){ return m.id === mId; });
    var mName    = mId === 'custom'
      ? document.getElementById('customMateriaName').value.trim()
      : (mInfo ? mInfo.name : mId);
    var mSsd     = mId === 'custom'
      ? (document.getElementById('customMateriaSsd').value.trim() || '')
      : (mInfo ? mInfo.ssd : '');
    var project  = createNewProject(pname, mId, mName, mSsd, wizardSelectedCdls.slice());
    closeModal('newProjectModal');
    showToast('✅ Progetto "' + pname + '" creato');
    openProject(project.id);
  }
}

function wizardBack() {
  if (wizardStep > 1) { wizardStep--; updateWizardUI(); }
}

// ════════════════════════════════════════════════
function autoSaveProject() {
  if (!CURRENT_PROJECT) return;
  try {
    if (CURRENT_PROJECT.framework) {
      var builderVisible =
        document.getElementById('builderContent').style.display !== 'none';
      if (builderVisible) {
        var v = function(id) {
          var el = document.getElementById(id);
          return (el && el.value.trim()) ? el.value.trim() : null;
        };
        CURRENT_PROJECT.framework.id               = v('fwId')             || CURRENT_PROJECT.framework.id;
        CURRENT_PROJECT.framework.name             = v('fwName')           || CURRENT_PROJECT.framework.name;
        CURRENT_PROJECT.framework.subject          = v('fwSubject')        || CURRENT_PROJECT.framework.subject;
        CURRENT_PROJECT.framework.context_strategy = v('fwContextStrategy') || CURRENT_PROJECT.framework.context_strategy || '';
      }
    }
    // Non salviamo i File object (non serializzabili) — solo metadati
    var toSave = Object.assign({}, CURRENT_PROJECT);
    toSave.files = (CURRENT_PROJECT.files || []).map(function(f) {
      return {
        name:    f.name,
        size:    f.size,
        status:  f.status,
        cdl:     f.cdl,
        materia: f.materia,
        args:    f.args || [],
        text:    ''   // non salviamo il testo grezzo per risparmiare spazio
      };
    });
    saveProject(toSave);
    CURRENT_PROJECT.updatedAt = toSave.updatedAt;
  } catch(e) {
    console.warn('autoSaveProject error:', e);
  }
}

// ════════════════════════════════════════════════
//  UPLOAD & DROP ZONE
// ════════════════════════════════════════════════
function setupDropZone() {
  var dz = document.getElementById('dropZone');
  if (!dz) return;
  dz.addEventListener('dragover', function(e){
    e.preventDefault(); dz.classList.add('drag-over');
  });
  dz.addEventListener('dragleave', function(){
    dz.classList.remove('drag-over');
  });
  dz.addEventListener('drop', function(e){
    e.preventDefault();
    dz.classList.remove('drag-over');
    handleFiles(e.dataTransfer.files);
  });
}

function handleFiles(fileList) {
  if (!CURRENT_PROJECT || !CURRENT_PROJECT.selectedCdl) {
    showToast('⚠️ Seleziona prima una classe di laurea', 'warning'); return;
  }
  var cdl     = CURRENT_PROJECT.selectedCdl;
  var materia = CURRENT_PROJECT.materiaId;
  Array.from(fileList).filter(function(f){ return f.name.endsWith('.pdf'); })
    .forEach(function(f) {
      if (CURRENT_PROJECT.files.find(function(x){
        return x.name === f.name && x.cdl === cdl;
      })) return;
      CURRENT_PROJECT.files.push({
        name: f.name, size: f.size, status: 'pending',
        cdl: cdl, materia: materia,
        file: f, text: '', args: []
      });
    });
  renderFileList();
  updateBtnProcess();
  updateUploadStats();
  document.getElementById('uploadStats').style.display = 'grid';
}

function renderFileList() {
  if (!CURRENT_PROJECT) return;
  var list     = document.getElementById('fileList');
  var cdl      = CURRENT_PROJECT.selectedCdl;
  var relevant = CURRENT_PROJECT.files.filter(function(f){
    return f.materia === CURRENT_PROJECT.materiaId;
  });
  if (!relevant.length) { list.innerHTML = ''; return; }
  list.innerHTML = relevant.map(function(f) {
    var icon =
      f.status==='pending'    ? '📄'
      : f.status==='processing' ? '⚙️'
      : f.status==='done'       ? '✅' : '❌';
    var statusLabel =
      f.status==='pending'    ? 'In attesa'
      : f.status==='processing' ? 'Elaborazione...'
      : f.status==='done'       ? 'Completato' : 'Errore';
    var sn  = f.name.replace(/'/g,"\\'");
    var sc  = f.cdl.replace(/'/g,"\\'");
    return '<div class="file-item ' + f.status + '">'
         + '  <div class="file-icon">' + icon + '</div>'
         + '  <div class="file-info">'
         + '    <div class="file-name">' + escHtml(f.name) + '</div>'
         + '    <div class="file-meta">'
         + '      ' + (f.size/1024).toFixed(0) + ' KB '
         + '      <span class="badge badge-blue">' + escHtml(f.cdl) + '</span>'
         + (f.args && f.args.length ? ' &nbsp;<strong>' + f.args.length + ' argomenti</strong>' : '')

async function startProcessing() {
  if (!SETTINGS.apiKey) {
    showToast('⚠️ Inserisci la chiave API nelle impostazioni', 'warning');
    document.getElementById('btnOpenSettings').click(); return;
  }
  if (!CURRENT_PROJECT) return;
  var pending = CURRENT_PROJECT.files.filter(function(f){
    return f.status==='pending' && f.materia===CURRENT_PROJECT.materiaId;
  });
  if (!pending.length) return;

  document.getElementById('processingProgress').style.display = 'block';
  document.getElementById('btnProcess').disabled = true;

  for (var i=0; i<pending.length; i++) {
    var f = pending[i];
    f.status = 'processing';
    renderFileList();
    updateProgress(i, pending.length, 'Elaborazione: ' + f.name);
    try {
      f.text = await extractTextFromPdf(f.file);
      f.args = await extractArgsWithAI(f.text, CURRENT_PROJECT.materiaId,
                                        CURRENT_PROJECT.materiaSsd, f.cdl);
      f.status = 'done';
      aggregateArgs(f);
    } catch(e) {
      f.status = 'error';
      console.error('Errore su', f.name, e);
    }
    renderFileList();
    updateUploadStats();
  }

  updateProgress(pending.length, pending.length, 'Elaborazione completata!');
  document.getElementById('countReview').textContent =
    CURRENT_PROJECT.files.filter(function(f){ return f.status==='done'; }).length;
  showToast('✅ Elaborati ' + pending.filter(function(f){ return f.status==='done'; }).length + ' PDF');
  updateReviewCdlFilter();
  autoSaveProject();
}

function updateProgress(done, total, label) {
  var pct = total>0 ? Math.round((done/total)*100) : 0;
  document.getElementById('progressLabel').textContent    = label;
  document.getElementById('progressPct').textContent      = pct + '%';
  document.getElementById('mainProgressFill').style.width = pct + '%';
}

async function extractTextFromPdf(file) {
  var buf  = await file.arrayBuffer();
  var pdf  = await pdfjsLib.getDocument({data:buf}).promise;
  var text = '';
  for (var p=1; p<=Math.min(pdf.numPages,12); p++) {
    var page    = await pdf.getPage(p);
    var content = await page.getTextContent();
    text += content.items.map(function(i){ return i.str; }).join(' ') + '\n';
  }
  return text.trim().substring(0,6000);
}

async function extractArgsWithAI(text, materiaId, materiaSsd, cdl) {
  var mInfo     = MATERIE.find(function(m){ return m.id===materiaId; });
  var mName     = mInfo ? mInfo.name : materiaId;
  var candidati = CANDIDATI[materiaId] || [];
  var ssd       = materiaSsd || (mInfo ? mInfo.ssd : '') || '';

  var systemPrompt =
    'Sei un esperto di didattica universitaria italiana specializzato nell\'analisi di programmi di insegnamento. '
    + 'Identifica gli argomenti disciplinari trattati, riconductoli alle categorie normalizzate fornite. '
    + 'Usa ESATTAMENTE il nome della categoria se l\'argomento corrisponde. '
    + 'Non inserire mai lo stesso argomento due volte. '
    + 'Rispondi SOLO con un array JSON di stringhe, senza spiegazioni.';

  var candidatiText = candidati.length
    ? 'CATEGORIE NORMALIZZATE:\n' + candidati.map(function(c,i){ return (i+1)+'. '+c; }).join('\n') + '\n\n'
    : '';

  var userPrompt =
    'Analizza questo programma universitario di ' + mName + ' (SSD: ' + ssd + ', CdL: ' + cdl + ').\n\n'
    + candidatiText
    + 'TESTO DEL PROGRAMMA:\n' + text + '\n\n'
    + 'Restituisci SOLO l\'array JSON degli argomenti trovati. Ogni argomento UNA SOLA VOLTA.';

  var resp = await fetch('https://api.openai.com/v1/chat/completions', {
    method: 'POST',
    headers: {'Content-Type':'application/json','Authorization':'Bearer '+SETTINGS.apiKey},
    body: JSON.stringify({
      model: SETTINGS.model,
      messages: [
        {role:'system', content:systemPrompt},
        {role:'user',   content:userPrompt}
      ],
      temperature: 0.1, max_tokens: 800
    })
  });
  if (!resp.ok) throw new Error('API error: ' + resp.status);
  var data  = await resp.json();
  var raw   = data.choices[0].message.content.trim();
  var parsed = [];
  try {
    var match = raw.match(/\[[\s\S]*\]/);
    if (match) parsed = JSON.parse(match[0]);
  } catch(e) {
    parsed = raw.split('\n').map(function(l){
      return l.replace(/^[-*\d.]\s*/,'').trim();
    }).filter(Boolean);
  }
  return [...new Set(parsed.map(function(a){ return a.trim(); }).filter(Boolean))];
}

function aggregateArgs(f) {
  if (!CURRENT_PROJECT) return;
  var m = CURRENT_PROJECT.materiaId;
  var c = f.cdl;
  if (!CURRENT_PROJECT.allArgs[m])         CURRENT_PROJECT.allArgs[m]         = {};
  if (!CURRENT_PROJECT.allArgs[m][c])      CURRENT_PROJECT.allArgs[m][c]      = {};
  if (!CURRENT_PROJECT.sampleCounts[m])    CURRENT_PROJECT.sampleCounts[m]    = {};
  if (!CURRENT_PROJECT.sampleCounts[m][c]) CURRENT_PROJECT.sampleCounts[m][c] = 0;
  CURRENT_PROJECT.sampleCounts[m][c]++;
  var unique = [...new Set((f.args||[]).map(function(a){ return a.trim(); }).filter(Boolean))];
  unique.forEach(function(arg) {
    if (!CURRENT_PROJECT.allArgs[m][c][arg]) CURRENT_PROJECT.allArgs[m][c][arg] = 0;
    CURRENT_PROJECT.allArgs[m][c][arg]++;
  });
}

function getThresholds(cdls, counts) {
  var minN = Math.min.apply(null, cdls.map(function(c){ return counts[c]||0; }));
  var corr = (SETTINGS.smallSampleCorrection==='conservative' && minN<30) ? 10 : 0;
  return { u: SETTINGS.threshUniversal-corr, p: SETTINGS.threshProfile-corr };
}

function getArgType(arg, freqsByCdl, cdls, counts) {
  var thresh = getThresholds(cdls, counts);
  var vals   = cdls.map(function(c){ return freqsByCdl[c]||0; });
  if (vals.every(function(v){ return v>=thresh.u; })) return 'universal';
  if (vals.some(function(v){  return v>=thresh.p; })) return 'profile_dependent';
  return 'discretionary';
}

function renderFrequencyTable() {
  if (!CURRENT_PROJECT) return;
  var materia = CURRENT_PROJECT.materiaId;
  var wrapper = document.getElementById('freqTableWrapper');
  var cdlData = (CURRENT_PROJECT.allArgs[materia])      || {};
  var counts  = (CURRENT_PROJECT.sampleCounts[materia]) || {};
  var cdls    = Object.keys(cdlData);

  if (!cdls.length) {
    wrapper.innerHTML='<div style="text-align:center;padding:30px;color:var(--gray-3)">Nessun dato disponibile. Elabora i PDF per procedere.</div>';
    return;
  }

  var allArgSet = new Set();
  cdls.forEach(function(cdl){ Object.keys(cdlData[cdl]).forEach(function(a){ allArgSet.add(a); }); });
  var args   = [...allArgSet].sort();
  var thresh = getThresholds(cdls, counts);

  var thead = '<tr><th>Argomento</th><th>Tipo</th>'
    + cdls.map(function(c){
        return '<th>' + escHtml(c) + '<br><span style="font-weight:400;font-size:11px">n=' + (counts[c]||0) + '</span></th>';
      }).join('')
    + '</tr>';

  var tbody = args.map(function(arg) {
    var freqsByCdl = {};
    cdls.forEach(function(cdl){
      var n   = counts[cdl]||1;
      var cnt = cdlData[cdl][arg]||0;
      freqsByCdl[cdl] = Math.round((cnt/n)*100);
    });
    var type = getArgType(arg, freqsByCdl, cdls, counts);
    var typeLabel =
      type==='universal'
        ? '<span class="badge badge-green">Universal</span>'
        : type==='profile_dependent'
        ? '<span class="badge badge-orange">Profile</span>'
        : '<span class="badge badge-gray">Discrezionale</span>';
    var cells = cdls.map(function(cdl){
      var n   = counts[cdl]||1;
      var cnt = cdlData[cdl][arg]||0;
      var pct = Math.round((cnt/n)*100);
      var bc  = pct>=thresh.u ? 'var(--green)' : pct>=thresh.p ? 'var(--orange)' : 'var(--gray-3)';
      return '<td><div class="freq-cell">'
           + '  <span class="freq-pct">' + pct + '%</span>'
           + '  <div style="flex:1;height:8px;border-radius:4px;background:var(--gray-2)">'
           + '    <div style="width:' + Math.max(pct,2) + '%;height:100%;border-radius:4px;background:' + bc + '"></div>'
           + '  </div>'
           + '  <span style="font-size:11px;color:var(--gray-3)">' + cnt + '/' + n + '</span>'
           + '</div></td>';
    }).join('');
    return '<tr><td><strong>' + escHtml(arg) + '</strong></td><td>' + typeLabel + '</td>' + cells + '</tr>';
  }).join('');

  wrapper.innerHTML = '<table class="freq-table"><thead>' + thead + '</thead><tbody>' + tbody + '</tbody></table>';

  // Confidence
  var minN     = Math.min.apply(null, cdls.map(function(c){ return counts[c]||0; }));
  var confCard = document.getElementById('confidenceCard');
  var confInfo = document.getElementById('confidenceInfo');
  confCard.style.display = 'block';
  if (minN < 30) {
    confInfo.innerHTML='<div class="callout warning"><span class="ci">⚠️</span><div>Campione piccolo: <strong>' + minN + ' programmi</strong>. Soglie abbassate del 10%.</div></div>';
  } else if (minN < 50) {
    confInfo.innerHTML='<div class="callout info"><span class="ci">ℹ️</span><div>Campione: <strong>' + minN + ' programmi</strong>. Affidabilità <strong>sufficiente</strong>.</div></div>';
  } else {
    confInfo.innerHTML='<div class="callout success"><span class="ci">✅</span><div>Campione <strong>solido</strong> (min ' + minN + ' programmi).</div></div>';
  }
}

function buildFramework() {
  if (!CURRENT_PROJECT) return;
  var materia = CURRENT_PROJECT.materiaId;
  var cdlData = CURRENT_PROJECT.allArgs[materia]      || {};
  var counts  = CURRENT_PROJECT.sampleCounts[materia] || {};
  var cdls    = Object.keys(cdlData);
  if (!cdls.length) { showToast('⚠️ Nessun dato di frequenza disponibile','warning'); return; }

  var allArgSet = new Set();
  cdls.forEach(function(cdl){ Object.keys(cdlData[cdl]).forEach(function(a){ allArgSet.add(a); }); });

  var modulesData = [...allArgSet].map(function(arg){
    var freqsByCdl = {};
    cdls.forEach(function(cdl){
      var n=counts[cdl]||1, cnt=cdlData[cdl][arg]||0;
      freqsByCdl[cdl] = Math.round((cnt/n)*100);
    });
    var type = getArgType(arg, freqsByCdl, cdls, counts);
    return {
      name:          arg,
      coverage_type: type,
      core_contents: [arg + ' — completare con argomenti specifici del modulo'],
      key_concepts:  [arg],
      _freqs:        freqsByCdl
    };
  });

  var order = {universal:0, profile_dependent:1, discretionary:2};
  modulesData.sort(function(a,b){ return order[a.coverage_type]-order[b.coverage_type]; });

  var programProfiles = cdls.map(function(cdl){
    var n          = counts[cdl]||0;
    var priModules = modulesData
      .filter(function(m){ return m.coverage_type==='universal'||(m._freqs[cdl]||0)>=SETTINGS.threshProfile; })
      .map(function(m){ return m.name; });
    var priText = priModules.slice(0,4).join(', ') + (priModules.length>4?'...':'');
    return {
      name: cdl,
      priority_modules: '[N=' + n + ' programmi] Focus su: ' + priText + '. Completare con AI.'
    };
  });

  CURRENT_PROJECT.framework = {
    id:               materia + '_framework',
    name:             CURRENT_PROJECT.materiaName || materia,
    subject:          materia,
    context_strategy: '',
    _meta:            { ssd: CURRENT_PROJECT.materiaSsd||'', cdls: cdls, counts: counts },
    content: {
      target_degrees:      cdls,
      evaluation_criteria: {
        contenuti_core:           'Quali argomenti fondamentali sono presenti/assenti?',
        approccio_metodologico:   'Approccio teorico/descrittivo vs empirico/applicativo?',
        orientamento_disciplinare:'Orientamento prevalente e scuola di pensiero?'
      },
      program_profiles: programProfiles,
      syllabus_modules: modulesData
    }
  };

  document.getElementById('exportContent').style.display = 'block';
  document.getElementById('exportEmpty').style.display   = 'none';
  autoSaveProject();
  showView('builderView');
  showToast('✅ Framework generato! Clicca "Genera con AI" per completarlo.');
}

async function generateWithAI() {

    var modules= fw.content.syllabus_modules;

    modules.forEach(function(m){
      var fl = cdls.map(function(c){
        return c + ':' + (m._freqs ? (m._freqs[c]!==undefined?m._freqs[c]:'?') : '?') + '%';

        discriminanti.push({name:m.name, delta:delta, vals:vals});
      }
    });
    discriminanti.sort(function(a,b){ return b.delta-a.delta; });
    discriminanti = discriminanti.slice(0,6);

    catch(e){ throw new Error('Step1 risposta non valida: '+raw1.substring(0,200)); }

        ? ['Comprende: '+m.fused_from.join(', ')]

      return {name:m.name,coverage_type:m.coverage_type,core_contents:cc,
              key_concepts:m.key_concepts||[m.name],_freqs:m.freqs||{}};
    });
    CURRENT_PROJECT.framework.content.syllabus_modules.sort(function(a,b){
      return ordMap[a.coverage_type]-ordMap[b.coverage_type];
    });
    renderModulesList();
    showToast('Step 1 OK — '+normModules.length+' moduli normalizzati');

    // STEP 2
    showToast('Step 2/2 - Generazione testi...','warning');
    var modNorm    = CURRENT_PROJECT.framework.content.syllabus_modules;
    var univStr    = modNorm.filter(function(m){return m.coverage_type==='universal';}).map(function(m){return m.name;}).join(', ')||'nessuno';

    var discStr    = modNorm.filter(function(m){return m.coverage_type==='discretionary';}).map(function(m){return m.name;}).join(', ')||'nessuno';

    modNorm.forEach(function(m){

    var modEnrTpl = modNorm.map(function(m,i){
      return '    {"name":"'+m.name+'","core_contents":"...","key_concepts":["...","...","..."]}' + (i<modNorm.length-1?',':'');
    }).join('\n');
    var profTpl = cdls.map(function(c,i){
      return '    {"name":"'+c+'","priority_modules":"..."}' + (i<cdls.length-1?',':'');
    }).join('\n');

    var step2System = 'Sei un consulente editoriale specializzato in testi universitari italiani. '

      + 'Scrivi in italiano professionale. Sii specifico, basati sui dati reali. '

function getFinalJson() {
  if (!CURRENT_PROJECT || !CURRENT_PROJECT.framework) return null;
  var fw = CURRENT_PROJECT.framework;

  var readField = function(id) {
    var el = document.getElementById(id);
    return (el && el.value.trim()) ? el.value.trim() : null;
  };

  var id               = readField('fwId')             || fw.id;
  var name             = readField('fwName')            || fw.name;
  var subjectRaw       = readField('fwSubject')         || fw.subject || '';
  var context_strategy = readField('fwContextStrategy') || fw.context_strategy || '';

  var subjectMap = {
    'psicologia generale':'psicologia','psicologia clinica':'psicologia',
    'psic. dello sviluppo':'psicologia','psicologia dello sviluppo':'psicologia',
    'psicologia sociale':'psicologia','psicometria':'psicologia',
    'antropologia culturale':'scienze_umane','sociologia generale':'scienze_umane',
    'chimica generale':'chimica','chimica organica':'chimica',
    'fisica':'fisica','fisica generale':'fisica',
    'matematica':'matematica','analisi matematica':'matematica',
    'microeconomia':'economia','macroeconomia':'economia','economia politica':'economia',
    'biologia':'biologia','biologia cellulare':'biologia','biologia molecolare':'biologia',
    'biochimica':'biologia','genetica':'biologia','microbiologia':'biologia','istologia':'istologia'
  };
  var subjectNormalized = subjectMap[subjectRaw.toLowerCase().trim()] || subjectRaw;

  var syllabus_modules = fw.content.syllabus_modules.map(function(m){
    var core_str = Array.isArray(m.core_contents)
      ? m.core_contents.filter(Boolean).join('. ')
      : (m.core_contents||'');
    var kc = (m.key_concepts||[]).map(function(k){
      return typeof k==='object'?(k.concetto||k.name||JSON.stringify(k)):String(k);
    }).filter(Boolean);
    return {name:m.name, core_contents:core_str, key_concepts:kc};
  });

  var evaluation_criteria = Object.assign({}, fw.content.evaluation_criteria||{});
  delete evaluation_criteria._note;

  return {
    id:id, name:name, subject:subjectNormalized, context_strategy:context_strategy,
    content:{
      target_degrees:   fw.content.target_degrees,
      evaluation_criteria: evaluation_criteria,
      program_profiles: fw.content.program_profiles.map(function(p){
        return {name:p.name, priority_modules:p.priority_modules};
      }),
      syllabus_modules: syllabus_modules
    }
  };
}

function downloadCsv() {
  if (!CURRENT_PROJECT) return;
  var materia = CURRENT_PROJECT.materiaId;
  if (!materia || !CURRENT_PROJECT.allArgs[materia]) {
    showToast('⚠️ Nessun dato frequenze da esportare','warning'); return;
  }
  var cdlData = CURRENT_PROJECT.allArgs[materia];
  var counts  = CURRENT_PROJECT.sampleCounts[materia]||{};
  var cdls    = Object.keys(cdlData);
  var allArgs = [...new Set(cdls.flatMap(function(c){ return Object.keys(cdlData[c]); }))].sort();
  var header  = ['Argomento','Tipo',...cdls.map(function(c){ return c+' (n='+(counts[c]||0)+') %'; })].join(',');
  var rows = allArgs.map(function(arg){
    var freqsByCdl={};
    cdls.forEach(function(cdl){
      var n=counts[cdl]||1;
      freqsByCdl[cdl]=Math.round(((cdlData[cdl][arg]||0)/n)*100);
    });
    var type=getArgType(arg,freqsByCdl,cdls,counts);
    return ['"'+arg+'"',type,...cdls.map(function(c){ return freqsByCdl[c]; })].join(',');
  });
  var csv  = [header,...rows].join('\n');
  var blob = new Blob([csv],{type:'text/csv;charset=utf-8;'});
  var a    = document.createElement('a');
  a.href   = URL.createObjectURL(blob);
  a.download = (CURRENT_PROJECT.name||materia).replace(/\s/g,'_')+'_frequenze.csv';
  document.body.appendChild(a); a.click();
  document.body.removeChild(a); URL.revokeObjectURL(a.href);
  showToast('✅ CSV scaricato');
}

function downloadReport() {
  autoSaveProject();
  var fw = getFinalJson();
  if (!fw) { showToast('⚠️ Genera prima un framework','warning'); return; }
  try {
    var jsPDF = window.jspdf.jsPDF;
    var doc   = new jsPDF({orientation:'portrait',unit:'mm',format:'a4'});
    var W=170, y=20;

    function addText(text,x,size,color,style){
      doc.setFontSize(size);
      doc.setTextColor.apply(doc,color||[0,0,0]);
      doc.setFont('helvetica',style==='bold'?'bold':'normal');
      var lines=doc.splitTextToSize(String(text||''),W-x+20);
      lines.forEach(function(line){
        if(y>275){doc.addPage();y=20;}
        doc.text(line,x,y); y+=size*0.45;
      }); y+=2;
    }
    function addSection(title){
      if(y>260){doc.addPage();y=20;} y+=3;
      doc.setFillColor(0,166,81);
      doc.roundedRect(20,y-4,W,8,1,1,'F');
      doc.setFontSize(10);doc.setTextColor(255,255,255);
      doc.setFont('helvetica','bold');
      doc.text(title,23,y+1); y+=9;
      doc.setFont('helvetica','normal');
    }

    doc.setFillColor(0,122,61); doc.rect(0,0,210,28,'F');
    doc.setFontSize(18);doc.setTextColor(255,255,255);doc.setFont('helvetica','bold');
    doc.text('MATRIX Framework Builder',20,12);
    doc.setFontSize(11);doc.setFont('helvetica','normal');
    doc.text('Report Metodologico — '+fw.name,20,20); y=36;

    addText('Progetto: '+(CURRENT_PROJECT?CURRENT_PROJECT.name:''),20,9,[100,100,100]);
    addText('ID: '+fw.id+'  |  Subject: '+fw.subject,20,9,[100,100,100]);
    addText('Generato: '+new Date().toLocaleDateString('it-IT')+'  |  Target: '+fw.content.target_degrees.join(', '),20,9,[100,100,100]);

    var counts2=CURRENT_PROJECT&&CURRENT_PROJECT.framework&&CURRENT_PROJECT.framework._meta
      ?CURRENT_PROJECT.framework._meta.counts:{};
    if(Object.keys(counts2).length){
      addSection('Campione analizzato');
      Object.entries(counts2).forEach(function(e){ addText(e[0]+': '+e[1]+' programmi',24,10); });
    }
    if(fw.context_strategy){
      addSection('Context Strategy');
      addText(fw.context_strategy,23,10);
    }
    if(fw.content.evaluation_criteria&&Object.keys(fw.content.evaluation_criteria).length){
      addSection('Criteri di valutazione diagnostica');
      Object.entries(fw.content.evaluation_criteria).forEach(function(e){
        addText(e[0].replace(/_/g,' '),23,10,[0,100,60],'bold');
        addText(e[1],27,9,[60,60,60]);
      });
    }
    if(fw.content.program_profiles&&fw.content.program_profiles.length){
      addSection('Program Profiles per CdL');
      fw.content.program_profiles.forEach(function(p){
        addText(p.name,23,11,[0,0,0],'bold');
        addText(p.priority_modules,25,9,[60,60,60]); y+=3;
      });
    }
    var internal=CURRENT_PROJECT&&CURRENT_PROJECT.framework
      ?CURRENT_PROJECT.framework.content.syllabus_modules:[];
    addSection('Syllabus Modules ('+fw.content.syllabus_modules.length+')');
    var nU=internal.filter(function(m){return m.coverage_type==='universal';}).length;
    var nP=internal.filter(function(m){return m.coverage_type==='profile_dependent';}).length;
    var nD=internal.filter(function(m){return m.coverage_type==='discretionary';}).length;
    addText('Universal: '+nU+'  |  Profile-dependent: '+nP+'  |  Discretionary: '+nD,23,9,[100,100,100]);
    y+=2;
    fw.content.syllabus_modules.forEach(function(m,i){
      if(y>268){doc.addPage();y=20;}
      var im=internal[i];
      var tipo=im&&im.coverage_type==='universal'?'[U]':im&&im.coverage_type==='profile_dependent'?'[P]':'[D]';
      var tc=im&&im.coverage_type==='universal'?[0,166,81]:im&&im.coverage_type==='profile_dependent'?[245,158,11]:[156,163,175];
      doc.setFontSize(10);doc.setFont('helvetica','bold');
      doc.setTextColor.apply(doc,tc); doc.text(tipo,23,y);
      doc.setTextColor(0,0,0); doc.text(m.name,31,y); y+=5;
      if(m.core_contents){
        var lines2=doc.splitTextToSize(m.core_contents,W-15);
        doc.setFontSize(8);doc.setFont('helvetica','normal');doc.setTextColor(80,80,80);
        lines2.slice(0,2).forEach(function(line){
          if(y>275){doc.addPage();y=20;}
          doc.text(line,31,y);y+=4;
        });
      } y+=1;
    });

    var tot=doc.internal.getNumberOfPages();
    for(var pg=1;pg<=tot;pg++){
      doc.setPage(pg);
      doc.setFontSize(8);doc.setTextColor(180,180,180);doc.setFont('helvetica','normal');
      doc.text('MATRIX Framework Builder  '+fw.name+'  pag. '+pg+'/'+tot,20,290);
    }
    doc.save(fw.id+'_report_'+new Date().toISOString().split('T')[0]+'.pdf');
    showToast('Report PDF scaricato');
  } catch(e) {
    console.error('downloadReport error:',e);
    showToast('Errore PDF: '+e.message,'error');
  }
}

function escHtml(str) {
  return String(str||'')
    .replace(/&/g,'&amp;').replace(/</g,'&lt;')
    .replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

function showToast(msg, type) {
  var t  = document.getElementById('toast');
  t.textContent = msg;
  t.style.background =
    type==='warning' ? '#92400e'
    : type==='error' ? '#991b1b'
    : '#166534';
  t.style.display = 'block';
  clearTimeout(t._timer);
  t._timer = setTimeout(function(){ t.style.display='none'; }, 3200);
}
