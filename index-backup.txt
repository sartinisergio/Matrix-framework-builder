  if (s.smallSampleCorrection) STATE.smallSampleCorrection = s.smallSampleCorrection;
  STATE.sessions = JSON.parse(localStorage.getItem('mfb_sessions') || '[]');

  // ── C2: Ripristino automatico STATE ──
  const savedState = localStorage.getItem('mfb_state');
  if (savedState) {
    try {
      const parsed = JSON.parse(savedState);
      if (parsed.allArgs)      STATE.allArgs      = parsed.allArgs;
      if (parsed.sampleCounts) STATE.sampleCounts = parsed.sampleCounts;
      if (parsed.framework)    STATE.framework    = parsed.framework;
      if (parsed.selectedMateria) STATE.selectedMateria = parsed.selectedMateria;
      if (parsed.selectedCdl)     STATE.selectedCdl     = parsed.selectedCdl;

      // Ripristina UI se ci sono dati
      if (parsed.allArgs && Object.keys(parsed.allArgs).length) {
        // Ricrea file fittizi per permettere updateFreqFilters()
        Object.keys(parsed.allArgs).forEach(materia => {
          Object.keys(parsed.allArgs[materia]).forEach(cdl => {
            if (!STATE.files.find(f => f.materia===materia && f.cdl===cdl)) {
              STATE.files.push({
                name: '_restored_' + materia + '_' + cdl,
                size: 0, status: 'done',
                cdl, materia, file: null, text: '', args: []
              });
            }
          });
        });
        updateFreqFilters();
        showToast('✅ Sessione precedente ripristinata');
      }
       // Ripristina builder se c'era un framework
      if (parsed.framework) {
        document.getElementById('exportContent').style.display = 'block';
        document.getElementById('exportEmpty').style.display   = 'none';
        setTimeout(() => {
          renderBuilder();
          showToast('✅ Framework precedente ripristinato');
        }, 100);
      }
    } catch(e) {
      console.warn('Impossibile ripristinare STATE:', e);
      localStorage.removeItem('mfb_state');
    }
  }
}

async function startProcessing() {
  if (!STATE.apiKey) {
    showToast('⚠️ Inserisci la chiave API nelle impostazioni', 'warning');
    openModal('settingsModal'); return;
  }
  const pending = STATE.files.filter(f =>
    f.status==='pending' &&
    f.materia===STATE.selectedMateria &&
    f.cdl===STATE.selectedCdl
  );
  if (!pending.length) return;

  document.getElementById('processingProgress').style.display = 'block';
  document.getElementById('btnProcess').disabled = true;

  for (let i=0; i<pending.length; i++) {
    const f = pending[i];
    f.status = 'processing';
    renderFileList();
    updateProgress(i, pending.length, `Elaborazione: ${f.name}`);
    try {
      f.text = await extractTextFromPdf(f.file);
      f.args = await extractArgsWithAI(f.text, f.materia, f.cdl);
      f.status = 'done';
      aggregateArgs(f);
    } catch(e) {
      f.status = 'error';
      console.error('Errore su', f.name, e);
    }
    renderFileList();
    updateUploadStats();
  }

  updateProgress(pending.length, pending.length, 'Elaborazione completata!');
  document.getElementById('countReview').textContent =
    STATE.files.filter(f=>f.status==='done').length;
  showToast(`✅ Elaborati ${pending.filter(f=>f.status==='done').length} PDF`);
  updateFreqFilters();
  saveSession();
}

function updateProgress(done, total, label) {
  const pct = total > 0 ? Math.round((done/total)*100) : 0;
  document.getElementById('progressLabel').textContent  = label;
  document.getElementById('progressPct').textContent    = pct+'%';
  document.getElementById('mainProgressFill').style.width = pct+'%';
}

// ── Estrazione testo da PDF ──
async function extractTextFromPdf(file) {
  const buf  = await file.arrayBuffer();
  const pdf  = await pdfjsLib.getDocument({data:buf}).promise;
  let text   = '';
  for (let p=1; p<=Math.min(pdf.numPages,12); p++) {
    const page    = await pdf.getPage(p);
    const content = await page.getTextContent();
    text += content.items.map(i=>i.str).join(' ') + '\n';
  }
  return text.trim().substring(0, 6000);
}

// ── Estrazione argomenti con AI ──
async function extractArgsWithAI(text, materiaId, cdl) {
  const materia    = MATERIE.find(m=>m.id===materiaId);
  const candidati  = CANDIDATI[materiaId] || [];
  const ssd        = materia?.ssd || '';

  const systemPrompt = `Sei un esperto di didattica universitaria italiana specializzato nell'analisi di programmi di insegnamento.
Il tuo compito è identificare gli argomenti disciplinari trattati in un programma universitario.
Devi ricondurre ogni argomento trovato alle categorie normalizzate fornite, usando ESATTAMENTE il nome della categoria se l'argomento corrisponde.
Se un argomento non corrisponde a nessuna categoria ma è rilevante per la disciplina, aggiungilo come nuova voce normalizzata.
IMPORTANTE: non inserire mai lo stesso argomento due volte — restituisci ogni argomento UNA SOLA VOLTA anche se appare più volte nel programma.
Rispondi SOLO con un array JSON di stringhe, senza spiegazioni. Esempio: ["Percezione visiva","Memoria di lavoro"]`;

  const userPrompt = `Analizza questo programma universitario di ${materia?.name||materiaId} (SSD: ${ssd}, Classe di laurea: ${cdl}).

CATEGORIE NORMALIZZATE DI RIFERIMENTO:
${candidati.map((c,i)=>`${i+1}. ${c}`).join('\n')}

TESTO DEL PROGRAMMA:
${text}

Identifica quali argomenti disciplinari sono presenti in questo programma.
Ricondili alle categorie normalizzate quando possibile.
Ogni argomento deve comparire UNA SOLA VOLTA nell'array.
Restituisci SOLO l'array JSON degli argomenti trovati.`;

  const resp = await fetch('https://api.openai.com/v1/chat/completions', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': 'Bearer ' + STATE.apiKey
    },
    body: JSON.stringify({
      model: STATE.model,
      messages: [
        { role:'system', content:systemPrompt },
        { role:'user',   content:userPrompt   }
      ],
      temperature: 0.1,
      max_tokens: 800
    })
  });

  if (!resp.ok) throw new Error('API error: ' + resp.status);
  const data = await resp.json();
  const raw  = data.choices[0].message.content.trim();

  // Parsing robusto + deduplicazione lato client
  let parsed = [];
  try {
    const match = raw.match(/\[[\s\S]*\]/);
    if (match) parsed = JSON.parse(match[0]);
  } catch(e) {
    parsed = raw.split('\n').map(l=>l.replace(/^[-*\d.]\s*/,'').trim()).filter(Boolean);
  }
  // Deduplicazione esplicita (per sicurezza)
  return [...new Set(parsed.map(a => a.trim()).filter(Boolean))];
}

// ── Aggregazione argomenti nella matrice (con protezione duplicati intra-file) ──
function aggregateArgs(f) {
  if (!STATE.allArgs[f.materia])           STATE.allArgs[f.materia]           = {};
  if (!STATE.allArgs[f.materia][f.cdl])    STATE.allArgs[f.materia][f.cdl]    = {};
  if (!STATE.sampleCounts[f.materia])      STATE.sampleCounts[f.materia]      = {};
  if (!STATE.sampleCounts[f.materia][f.cdl]) STATE.sampleCounts[f.materia][f.cdl] = 0;

  STATE.sampleCounts[f.materia][f.cdl]++;

  // Usa un Set per garantire che ogni argomento sia contato UNA SOLA VOLTA per file
  const uniqueArgs = [...new Set(f.args.map(a => a.trim()).filter(Boolean))];
  uniqueArgs.forEach(arg => {
    if (!STATE.allArgs[f.materia][f.cdl][arg])
      STATE.allArgs[f.materia][f.cdl][arg] = 0;
    STATE.allArgs[f.materia][f.cdl][arg]++;
  });
}

// ════════════════════════════════════════════════
//  FREQUENZE
// ════════════════════════════════════════════════
function getThresholds(cdls, counts) {
  const minN = Math.min(...cdls.map(c => counts[c]||0));
  const corr = STATE.smallSampleCorrection==='conservative' && minN < 30 ? 10 : 0;
  return {
    u: STATE.threshUniversal - corr,
    p: STATE.threshProfile   - corr
  };
}

function getArgType(arg, freqsByCdl, cdls, counts) {
  const thresh = getThresholds(cdls, counts);
  const vals   = cdls.map(c => freqsByCdl[c] || 0);
  if (vals.every(v => v >= thresh.u)) return 'universal';
  if (vals.some(v  => v >= thresh.p)) return 'profile_dependent';
  return 'discretionary';
}

function renderFrequencyTable() {
  const materia = document.getElementById('freqMateriaFilter').value;
  if (!materia) return;

  const wrapper = document.getElementById('freqTableWrapper');
  const cdlData = STATE.allArgs[materia]      || {};
  const counts  = STATE.sampleCounts[materia] || {};
  const cdls    = Object.keys(cdlData);

  if (!cdls.length) {
    wrapper.innerHTML = '<div style="text-align:center;padding:30px;color:var(--gray-3)">Nessun dato per questa materia.</div>';
    return;
  }

  const allArgSet = new Set();
  cdls.forEach(cdl => Object.keys(cdlData[cdl]).forEach(a => allArgSet.add(a)));
  const args = [...allArgSet].sort();

  const thresh = getThresholds(cdls, counts);

  const thead = `<tr>
    <th>Argomento</th><th>Tipo</th>
    ${cdls.map(c=>`<th>${c}<br><span style="font-weight:400;font-size:11px">n=${counts[c]||0}</span></th>`).join('')}
  </tr>`;

  const tbody = args.map(arg => {
    const freqsByCdl = Object.fromEntries(
      cdls.map(cdl => {
        const n   = counts[cdl] || 1;
        const cnt = cdlData[cdl][arg] || 0;
        return [cdl, Math.round((cnt/n)*100)];
      })
    );
    const type = getArgType(arg, freqsByCdl, cdls, counts);
    const typeLabel =
      type==='universal'        ? '<span class="badge badge-green">Universal</span>'
      : type==='profile_dependent' ? '<span class="badge badge-orange">Profile</span>'
      :                              '<span class="badge badge-gray">Discrezionale</span>';

    const cells = cdls.map(cdl => {
      const n   = counts[cdl] || 1;
      const cnt = cdlData[cdl][arg] || 0;
      const pct = Math.round((cnt/n)*100);
      const barColor = pct>=thresh.u?'var(--green)':pct>=thresh.p?'var(--orange)':'var(--gray-3)';
      return `<td><div class="freq-cell">
        <span class="freq-pct">${pct}%</span>
        <div style="flex:1;height:8px;border-radius:4px;background:var(--gray-2)">
          <div style="width:${Math.max(pct,2)}%;height:100%;border-radius:4px;background:${barColor}"></div>
        </div>
        <span style="font-size:11px;color:var(--gray-3)">${cnt}/${n}</span>
      </div></td>`;
    }).join('');

    return `<tr><td><strong>${arg}</strong></td><td>${typeLabel}</td>${cells}</tr>`;
  }).join('');

  wrapper.innerHTML = `<table class="freq-table"><thead>${thead}</thead><tbody>${tbody}</tbody></table>`;

  // Confidence card
  const minN     = Math.min(...cdls.map(c=>counts[c]||0));
  const confCard = document.getElementById('confidenceCard');
  const confInfo = document.getElementById('confidenceInfo');
  confCard.style.display = 'block';
  if (minN < 30) {
    confInfo.innerHTML = `<div class="callout warning"><span class="ci">⚠️</span>
      <div>Il campione più piccolo ha <strong>${minN} programmi</strong>.
      Soglie abbassate automaticamente del 10%. Interpretare con cautela.</div></div>`;
  } else if (minN < 50) {
    confInfo.innerHTML = `<div class="callout info"><span class="ci">ℹ️</span>
      <div>Campione più piccolo: <strong>${minN} programmi</strong>.
      Affidabilità <strong>sufficiente</strong> — pattern principali robusti.</div></div>`;
  } else {
    confInfo.innerHTML = `<div class="callout success"><span class="ci">✅</span>
      <div>Campione <strong>solido</strong> (min ${minN} programmi).
      Frequenze ad alta affidabilità statistica.</div></div>`;
  }
}

function updateCriterionKey(idx, newKey) {
  if (!STATE.framework) return;

  const newCriteria   = {};
  entries.forEach(([k, v], i) => {
    newCriteria[i === idx ? newKey : k] = v;
  });
  STATE.framework.content.evaluation_criteria = newCriteria;
  refreshJsonPreview();
}

/**
 * buildFramework()
 * Costruisce STATE.framework con la struttura interna del Builder.
 * I moduli hanno core_contents come ARRAY (per UI granulare).
 * L'export converte in stringa (getFinalJson).
 *
 * Struttura interna STATE.framework:
 * {
 *   id, name, subject, context_strategy,
 *   _meta: { ssd, cdls, counts },   ← solo uso interno, non esportato
 *   content: {
 *     target_degrees: [],
 *     evaluation_criteria: {},       ← generato dall'AI in Fase B, per ora skeleton
 *     program_profiles: [
 *       { name, priority_modules }   ← priority_modules = STRINGA narrativa
 *     ],
 *     syllabus_modules: [
 *       { name, coverage_type,       ← coverage_type = solo interno Builder
 *         core_contents: [],         ← ARRAY internamente
 *         key_concepts: [],
 *         _freqs: {}                 ← frequenze per CdL, solo interno
 *       }
 *     ]
 *   }
 * }
 */
function buildFramework() {
  const materia = document.getElementById('freqMateriaFilter').value;
  if (!materia) { showToast('⚠️ Seleziona prima una materia', 'warning'); return; }

  const cdlData = STATE.allArgs[materia]      || {};
  const counts  = STATE.sampleCounts[materia] || {};
  const cdls    = Object.keys(cdlData);
  const info    = MATERIE.find(m => m.id===materia);

  if (!cdls.length) {
    showToast('⚠️ Nessun dato di frequenza disponibile', 'warning'); return;
  }

  // ── Raccogli tutti gli argomenti con frequenze ──
  const allArgSet = new Set();
  cdls.forEach(cdl => Object.keys(cdlData[cdl]).forEach(a => allArgSet.add(a)));

  const modulesData = [...allArgSet].map(arg => {
    const freqsByCdl = Object.fromEntries(
      cdls.map(cdl => {
        const n   = counts[cdl] || 1;
        const cnt = cdlData[cdl][arg] || 0;
        return [cdl, Math.round((cnt/n)*100)];
      })
    );
    const type = getArgType(arg, freqsByCdl, cdls, counts);
    return {
      name: arg,
      coverage_type: type,
      // core_contents come array (interno Builder) — l'export convertirà in stringa
      core_contents: [arg + ' — completare con argomenti specifici del modulo'],
      key_concepts:  [arg],
      _freqs: freqsByCdl
    };
  });

  // ── Ordina: universal → profile_dependent → discretionary ──
  const order = { universal:0, profile_dependent:1, discretionary:2 };
  modulesData.sort((a,b) => order[a.coverage_type] - order[b.coverage_type]);

  // ── Program profiles: testo narrativo per CdL ──
  const cdlNames = {
    'L-24':         'L-24 Scienze e Tecniche Psicologiche',
    'L-19/LM-85bis':'L-19/LM-85bis Scienze dell\'Educazione e della Formazione Primaria',
    'ALTRO':        'Altra classe di laurea'
  };

  const programProfiles = cdls.map(cdl => {
    const n          = counts[cdl] || 0;
    const priModules = modulesData
      .filter(m => m.coverage_type==='universal' || (m._freqs[cdl]||0) >= STATE.threshProfile)
      .map(m => m.name);
    const secModules = modulesData
      .filter(m => m.coverage_type==='profile_dependent' && (m._freqs[cdl]||0) >= 30 && (m._freqs[cdl]||0) < STATE.threshProfile)
      .map(m => m.name);

    // Testo narrativo base (da completare manualmente o con AI in Fase B)
    const priText = priModules.slice(0,4).join(', ') + (priModules.length>4 ? '...' : '');
    const priorityNarrative =
      `[N=${n} programmi analizzati] Focus su: ${priText}. ` +
      (secModules.length
        ? `Secondari: ${secModules.slice(0,3).join(', ')}. `
        : '') +
      `— Completare con la logica didattica specifica per ${cdlNames[cdl]||cdl}.`;

    return {
      name: cdlNames[cdl] || cdl,
      priority_modules: priorityNarrative
    };
  });

  // ── Evaluation criteria: skeleton (sarà generato da AI in Fase B) ──
  const evaluationCriteria = {
    _note: 'Completare con i criteri diagnostici specifici per ' + (info?.name||materia),
    contenuti_core: 'Check: quali argomenti fondamentali sono presenti/assenti?',
    approccio_metodologico: 'Verificare: approccio teorico/descrittivo vs empirico/applicativo?',
    orientamento_disciplinare: 'Identificare: orientamento prevalente e scuola di pensiero?'
  };

  // ── Assembla framework ──
  STATE.framework = {
    id:               materia + '_framework',
    name:             info?.name || materia,
    subject:          info?.name || materia,
    context_strategy: '',
    // Campi interni al Builder (non esportati nel JSON finale)
    _meta: {
      ssd:    info?.ssd || '',
      cdls:   cdls,
      counts: counts
    },
    content: {
      target_degrees:      cdls,
      evaluation_criteria: evaluationCriteria,
      program_profiles:    programProfiles,
      syllabus_modules:    modulesData
    }
  };

 showView('builderView');
  renderBuilder();
  document.getElementById('exportContent').style.display = 'block';
  document.getElementById('exportEmpty').style.display   = 'none';
  autoSaveState();
  showToast('✅ Framework generato! Completa i dettagli nel Builder.');
}

/**
 * getFinalJson()
 * Produce il JSON finale compatibile con MATRIX v1.2.0.
 * - Rimuove tutti i campi interni (_meta, _freqs, coverage_type)
 * - Converte core_contents da array a stringa narrativa
 * - Normalizza key_concepts come array di stringhe semplici
 */
function getFinalJson() {
  if (!STATE.framework) return null;
  const fw = STATE.framework;

  // Leggi dai campi HTML se disponibili, altrimenti usa STATE
  const id               = document.getElementById('fwId')?.value.trim()              || fw.id;
  const name             = document.getElementById('fwName')?.value.trim()            || fw.name;
  const subject          = document.getElementById('fwSubject')?.value.trim()         || fw.subject;
  const context_strategy = document.getElementById('fwContextStrategy')?.value.trim() || fw.context_strategy;

  const syllabus_modules = fw.content.syllabus_modules.map(m => {
    const core_str = Array.isArray(m.core_contents)
      ? m.core_contents.filter(Boolean).join('. ')
      : (m.core_contents || '');
    const kc = (m.key_concepts || []).map(k =>
      typeof k === 'object' ? (k.concetto || k.name || JSON.stringify(k)) : String(k)
    ).filter(Boolean);
    return { name: m.name, core_contents: core_str, key_concepts: kc };
  });

  const evaluation_criteria = { ...fw.content.evaluation_criteria };
  delete evaluation_criteria._note;

  const result = {
    id, name, subject, context_strategy,
    content: {
      target_degrees:      fw.content.target_degrees,
      evaluation_criteria,
      program_profiles: fw.content.program_profiles.map(p => ({
        name:             p.name,
        priority_modules: p.priority_modules
      })),
      syllabus_modules
    }
  };

  // Aggiorna la cache nello STATE
  fw._cachedJson = result;
  return result;
}

function refreshJsonPreview() {
  const fw  = getFinalJson();
  const el  = document.getElementById('jsonPreview');
  if (!fw || !el) return;
  el.textContent = JSON.stringify(fw, null, 2);
}

// ── Download JSON ──
function downloadJson() {
  const fw = getFinalJson();
  if (!fw) { showToast('⚠️ Nessun framework da esportare', 'warning'); return; }
  const blob = new Blob([JSON.stringify(fw, null, 2)], {type:'application/json'});
  const a    = document.createElement('a');
  a.href     = URL.createObjectURL(blob);
  a.download = fw.id + '.json';
  a.click();
  showToast('✅ JSON scaricato: ' + fw.id + '.json');
}

function copyJsonClipboard() {
  const fw = getFinalJson();
  if (!fw) return;
  navigator.clipboard.writeText(JSON.stringify(fw, null, 2))
    .then(()  => showToast('✅ JSON copiato negli appunti'))
    .catch(()  => showToast('❌ Errore copia', 'error'));
}

// ── Download CSV frequenze ──
function downloadCsv() {
  const materia = document.getElementById('freqMateriaFilter').value;
  if (!materia || !STATE.allArgs[materia]) {
    showToast('⚠️ Nessun dato frequenze da esportare', 'warning'); return;
  }
  const cdlData = STATE.allArgs[materia];
  const counts  = STATE.sampleCounts[materia] || {};
  const cdls    = Object.keys(cdlData);
  const allArgs = [...new Set(cdls.flatMap(c=>Object.keys(cdlData[c])))].sort();

  const header = ['Argomento','Tipo',
    ...cdls.map(c=>`${c} (n=${counts[c]||0}) %`)].join(',');

  const rows = allArgs.map(arg => {
    const freqsByCdl = Object.fromEntries(
      cdls.map(cdl => {
        const n=counts[cdl]||1;
        return [cdl, Math.round(((cdlData[cdl][arg]||0)/n)*100)];
      })
    );
    const type = getArgType(arg, freqsByCdl, cdls, counts);
    return [`"${arg}"`, type, ...cdls.map(c=>freqsByCdl[c])].join(',');
  });

  const csv  = [header, ...rows].join('\n');
  const blob = new Blob([csv], {type:'text/csv'});
  const a    = document.createElement('a');
  a.href     = URL.createObjectURL(blob);
  a.download = materia + '_frequenze.csv';
  a.click();
  showToast('✅ CSV scaricato');
}

function exportFreqExcel() { downloadCsv(); }

// ── Report PDF ──
function downloadReport() {
  // Usa la cache se disponibile (quando chiamato da View 5)
  const fw = getFinalJson() || STATE.framework?._cachedJson;
  if (!fw) { showToast('⚠️ Genera prima un framework', 'warning'); return; }
  try {
    const { jsPDF } = window.jspdf;
    const doc = new jsPDF({ orientation:'portrait', unit:'mm', format:'a4' });
    const W   = 170; // larghezza utile
    let y     = 20;

    // ── Utility ──
    function addText(text, x, size, color, style) {
      doc.setFontSize(size);
      doc.setTextColor(...(color||[0,0,0]));
      if (style==='bold') doc.setFont('helvetica','bold');
      else doc.setFont('helvetica','normal');
      const lines = doc.splitTextToSize(String(text), W - x + 20);
      lines.forEach(line => {
        if (y > 275) { doc.addPage(); y = 20; }
        doc.text(line, x, y);
        y += size * 0.45;
      });
      y += 2;
    }

    function addSeparator(color) {
      if (y > 275) { doc.addPage(); y = 20; }
      doc.setDrawColor(...(color||[0,166,81]));
      doc.setLineWidth(0.4);
      doc.line(20, y, 190, y);
      y += 5;
    }

    function addSection(title) {
      if (y > 260) { doc.addPage(); y = 20; }
      y += 3;
      doc.setFillColor(0, 166, 81);
      doc.roundedRect(20, y-4, W, 8, 1, 1, 'F');
      doc.setFontSize(10); doc.setTextColor(255,255,255);
      doc.setFont('helvetica','bold');
      doc.text(title, 23, y+1);
      y += 9;
      doc.setFont('helvetica','normal');
    }

    // ── INTESTAZIONE ──
    doc.setFillColor(0, 122, 61);
    doc.rect(0, 0, 210, 28, 'F');
    doc.setFontSize(18); doc.setTextColor(255,255,255);
    doc.setFont('helvetica','bold');
    doc.text('MATRIX Framework Builder', 20, 12);
    doc.setFontSize(11); doc.setFont('helvetica','normal');
    doc.text('Report Metodologico — ' + fw.name, 20, 20);
    y = 36;

    // ── METADATI ──
    addText('ID: ' + fw.id, 20, 9, [100,100,100]);
    addText('Materia: ' + fw.name + '  |  Subject: ' + fw.subject, 20, 9, [100,100,100]);
    addText('Generato il: ' + new Date().toLocaleDateString('it-IT') +
      '  |  Target: ' + fw.content.target_degrees.join(', '), 20, 9, [100,100,100]);
    addSeparator();

    // ── CAMPIONE ──
    const counts = STATE.framework?._meta?.counts || {};
    if (Object.keys(counts).length) {
      addSection('📊 Campione analizzato');
      Object.entries(counts).forEach(([cdl, n]) => {
        addText(`${cdl}: ${n} programmi universitari`, 24, 10);
      });
    }

    // ── CONTEXT STRATEGY ──
    if (fw.context_strategy) {
      addSection('🎯 Context Strategy');
      addText(fw.context_strategy, 23, 10);
    }

    // ── EVALUATION CRITERIA ──
    if (fw.content.evaluation_criteria &&
        Object.keys(fw.content.evaluation_criteria).length) {
      addSection('🔍 Criteri di valutazione diagnostica');
      Object.entries(fw.content.evaluation_criteria).forEach(([key, val]) => {
        addText('• ' + key.replace(/_/g,' '), 23, 10, [0,100,60], 'bold');
        addText(val, 27, 9, [60,60,60]);
      });
    }

    // ── PROGRAM PROFILES ──
    if (fw.content.program_profiles?.length) {
      addSection('🎓 Program Profiles per CdL');
      fw.content.program_profiles.forEach(p => {
        addText(p.name, 23, 11, [0,0,0], 'bold');
        addText(p.priority_modules, 25, 9, [60,60,60]);
        y += 3;
      });
    }

    // ── SYLLABUS MODULES ──
    addSection('📚 Syllabus Modules (' + fw.content.syllabus_modules.length + ')');

    // Contatori per tipo (dall'STATE interno)
    const internal = STATE.framework?.content?.syllabus_modules || [];
    const nUni = internal.filter(m=>m.coverage_type==='universal').length;
    const nPro = internal.filter(m=>m.coverage_type==='profile_dependent').length;
    const nDis = internal.filter(m=>m.coverage_type==='discretionary').length;
    addText(
      `Universal: ${nUni}  |  Profile-dependent: ${nPro}  |  Discretionary: ${nDis}`,
      23, 9, [100,100,100]
    );
    y += 2;

    fw.content.syllabus_modules.forEach((m, i) => {
      if (y > 268) { doc.addPage(); y = 20; }
      const internalM = internal[i];
      const tipo =
        internalM?.coverage_type==='universal'         ? '[U]'
        : internalM?.coverage_type==='profile_dependent' ? '[P]'
        : '[D]';
      const tipoColor =
        internalM?.coverage_type==='universal'         ? [0,166,81]
        : internalM?.coverage_type==='profile_dependent' ? [245,158,11]
        : [156,163,175];

      doc.setFontSize(10); doc.setFont('helvetica','bold');
      doc.setTextColor(...tipoColor);
      doc.text(tipo, 23, y);
      doc.setTextColor(0,0,0);
      doc.text(m.name, 31, y);
      y += 5;

      if (m.core_contents) {
        const lines = doc.splitTextToSize(m.core_contents, W - 15);
        doc.setFontSize(8); doc.setFont('helvetica','normal');
        doc.setTextColor(80,80,80);
        lines.slice(0,2).forEach(line => {
          if (y > 275) { doc.addPage(); y = 20; }
          doc.text(line, 31, y); y += 4;
        });
      }
      y += 1;
    });

    // ── FOOTER su ogni pagina ──
    const totalPages = doc.internal.getNumberOfPages();
    for (let p=1; p<=totalPages; p++) {
      doc.setPage(p);
      doc.setFontSize(8); doc.setTextColor(180,180,180);
      doc.setFont('helvetica','normal');
      doc.text(
        'MATRIX Framework Builder — ' + fw.name + ' — pag. ' + p + '/' + totalPages,
        20, 290
      );
    }

    doc.save(fw.id + '_report_' +
      new Date().toISOString().split('T')[0] + '.pdf');
    showToast('✅ Report PDF scaricato');

  } catch(e) {
    showToast('❌ Errore generazione PDF', 'error');
    console.error(e);
  }
}

function saveSession() {
  const session = {
    id:          Date.now(),
    date:        new Date().toLocaleDateString('it-IT'),
    materie:     [...new Set(STATE.files.map(f=>f.materia))],
    totFiles:    STATE.files.filter(f=>f.status==='done').length,
    allArgs:     STATE.allArgs,
    sampleCounts:STATE.sampleCounts
  };
  STATE.sessions.unshift(session);
  if (STATE.sessions.length > 10) STATE.sessions.pop();
  localStorage.setItem('mfb_sessions', JSON.stringify(STATE.sessions));
  renderSessions();
}

function renderSessions() {
  const list = document.getElementById('sessionsList');
  if (!STATE.sessions.length) {
    list.innerHTML = `<div style="text-align:center;padding:40px;color:var(--gray-3)">
      <i class="fas fa-history" style="font-size:32px;display:block;margin-bottom:8px"></i>
      <p>Nessuna sessione salvata.</p></div>`;
    return;
  }
  list.innerHTML = STATE.sessions.map(s => `
    <div style="display:flex;align-items:center;gap:12px;padding:14px;border-bottom:1px solid var(--gray-2)">
      <div style="flex:1">
        <div style="font-weight:700;font-size:13px">${s.date}</div>
        <div style="font-size:12px;color:var(--text-2)">
          ${s.materie.join(', ')} · ${s.totFiles} PDF elaborati
        </div>
      </div>
      <button class="btn btn-secondary btn-sm" onclick="loadSession(${s.id})">
        <i class="fas fa-folder-open"></i> Riprendi
      </button>
      <button class="btn btn-danger btn-sm" onclick="deleteSession(${s.id})">
        <i class="fas fa-trash"></i>
      </button>
    </div>
  `).join('');
}

function loadSession(id) {
  const s = STATE.sessions.find(x=>x.id===id);
  if (!s) return;
  STATE.allArgs      = s.allArgs;
  STATE.sampleCounts = s.sampleCounts;
  updateFreqFilters();
  showView('frequencyView');
  showToast('✅ Sessione ripresa');
}

function deleteSession(id) {
  STATE.sessions = STATE.sessions.filter(s=>s.id!==id);
  localStorage.setItem('mfb_sessions', JSON.stringify(STATE.sessions));
  renderSessions();
}
